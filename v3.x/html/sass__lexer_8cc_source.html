02784              * treat this as a final EOF.

02786             return EOB_ACT_END_OF_FILE;

02811

02815             YY_CURRENT_BUFFER_LVALUE->yy_buf_size - number_to_move - 1;

02845                 "fatal error - scanner input buffer overflow" );

02848 
02849             num_to_read = YY_CURRENT_BUFFER_LVALUE->yy_buf_size -
02850                         number_to_move - 1;
02851 
02852             }
02853

02855             num_to_read = YY_READ_BUF_SIZE;
02856 
02857         /* Read in more data. */
02858         YY_INPUT( (&YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[number_to_move]),
02859             (yy_n_chars), (size_t) num_to_read );
02860 
02861         YY_CURRENT_BUFFER_LVALUE->yy_n_chars = (yy_n_chars);
02862         }
02863 
02864     if ( (yy_n_chars) == 0 )
02865         {
02866         if ( number_to_move == YY_MORE_ADJ )
02867             {
02868             ret_val = EOB_ACT_END_OF_FILE;
02869             sass_restart(sass_in  );
02870             }
02871 
02872         else
02873             {
02874             ret_val = EOB_ACT_LAST_MATCH;
02875             YY_CURRENT_BUFFER_LVALUE->yy_buffer_status =
02876                 YY_BUFFER_EOF_PENDING;
02877             }
02878         }
02879 
02880     else
02881         ret_val = EOB_ACT_CONTINUE_SCAN;
02882 
02883     if ((yy_size_t) ((yy_n_chars) + number_to_move) > YY_CURRENT_BUFFER_LVALUE->yy_buf_size) {

02885         yy_size_t new_size = (yy_n_chars) + number_to_move + ((yy_n_chars) >> 1);
02886         YY_CURRENT_BUFFER_LVALUE->yy_ch_buf = (char *) sass_realloc((void *) YY_CURRENT_BUFFER_LVALUE->yy_ch_buf,new_size  );

02888             YY_FATAL_ERROR( "out of dynamic memory in yy_get_next_buffer()" );
02889     }
02890 
02891     (yy_n_chars) += number_to_move;
02892     YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[(yy_n_chars)] = YY_END_OF_BUFFER_CHAR;
02893     YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[(yy_n_chars) + 1] = YY_END_OF_BUFFER_CHAR;
02894 
02895     (yytext_ptr) = &YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[0];
02896 
02897     return ret_val;
