00071 void CMesh::_ComputeSize( const Configuration &config ) {
00072 
00073   int k = config.GetInt( "k" );
00074   int n = config.GetInt( "n" );
00075   assert(n <= 2); // broken for n > 2
00076   int c = config.GetInt( "c" );
00077   assert(c == 4); // broken for c != 4
00078 
00079   ostringstream router_name;
00080   //how many routers in the x or y direction
00081   _xcount = config.GetInt("x");
00082   _ycount = config.GetInt("y");
00083   assert(_xcount == _ycount); // broken for asymmetric topologies
00084   //configuration of hohw many clients in X and Y per router
00085   _xrouter = config.GetInt("xr");
00086   _yrouter = config.GetInt("yr");
00087   assert(_xrouter == _yrouter); // broken for asymmetric concentration
00088 
00089   gK = _k = k ;
00090   gN = _n = n ;
00091   gC = _c = c ;
00092 
00093   assert(c == _xrouter*_yrouter);
00094   
00095   _nodes    = _c * powi( _k, _n); // Number of nodes in network
00096   _size     = powi( _k, _n);      // Number of routers in network
00097   _channels = 2 * _n * _size;     // Number of channels in network
00098 
00099   _cX = _c / _n ;   // Concentration in X Dimension 
00100   _cY = _c / _cX ;  // Concentration in Y Dimension
00101 
00102   //
00103   _memo_NodeShiftX = _cX >> 1 ;
00104   _memo_NodeShiftY = log_two(gK * _cX) + ( _cY >> 1 ) ;
00105   _memo_PortShiftY = log_two(gK * _cX)  ;
00106 
00107 }
