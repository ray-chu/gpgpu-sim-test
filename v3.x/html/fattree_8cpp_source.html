00069 void FatTree::_ComputeSize( const Configuration& config )
00070 {
00071 
00072   _k = config.GetInt( "k" );
00073   _n = config.GetInt( "n" );
00074    
00075   gK = _k; gN = _n;
00076   
00077   _nodes = powi( _k, _n );
00078 
00079   //levels * routers_per_level
00080   _size = _n * powi( _k , _n - 1 );
00081 
00082   //(channels per level = k*routers_per_level* up/down) * (levels-1)
00083   _channels = (2*_k * powi( _k , _n-1 ))*(_n-1);
