00087 void MaxSizeMatch::Allocate( )
00088 {
00089 
00090   // augment as many times as possible 
00091   // (this is an O(N^3) maximum-size matching algorithm)
00092   while( _ShortestAugmenting( ) );
00093 
00094   // next time, start at next input to ensure fairness
00095   _prio = (_prio + 1) % _inputs;
