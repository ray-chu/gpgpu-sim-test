00101 void Stats::AddSample( double val )
00102 {
00103   ++_num_samples;
00104   _sample_sum += val;
00105 
00106   // NOTE: the negation ensures that NaN values are handled correctly!
00107   _max = !(val <= _max) ? val : _max;
00108   _min = !(val >= _min) ? val : _min;
00109 
00110   //double clamp between 0 and num_bins-1
00111   int b = (int)fmax(floor( val / _bin_size ), 0.0);
00112   b = (b >= _num_bins) ? (_num_bins - 1) : b;
00113 
00114   _hist[b]++;
