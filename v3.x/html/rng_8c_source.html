00055 #else
00056 void ran_start(seed)    /* do this before using ran_array */
00057   long seed;            /* selector for different streams */
00058 #endif
00059 {
00060   register int t,j;
00061   long x[KK+KK-1];              /* the preparation buffer */
00062   register long ss=(seed+2)&(MM-2);
00063   for (j=0;j<KK;j++) {
00064     x[j]=ss;                      /* bootstrap the buffer */
00065     ss<<=1; if (ss>=MM) ss-=MM-2; /* cyclic shift 29 bits */
00066   }
00067   x[1]++;              /* make x[1] (and only x[1]) odd */
00068   for (ss=seed&(MM-1),t=TT-1; t; ) {       
00069     for (j=KK-1;j>0;j--) x[j+j]=x[j], x[j+j-1]=0; /* "square" */
00070     for (j=KK+KK-2;j>=KK;j--)
00071       x[j-(KK-LL)]=mod_diff(x[j-(KK-LL)],x[j]),
00072       x[j-KK]=mod_diff(x[j-KK],x[j]);
00073     if (is_odd(ss)) {              /* "multiply by z" */
00074       for (j=KK;j>0;j--)  x[j]=x[j-1];
00075       x[0]=x[KK];            /* shift the buffer cyclically */
00076       x[LL]=mod_diff(x[LL],x[KK]);
00077     }
00078     if (ss) ss>>=1; else t--;
00079   }
00080   for (j=0;j<LL;j++) ran_x[j+KK-LL]=x[j];
00081   for (;j<KK;j++) ran_x[j-LL]=x[j];
00082   for (j=0;j<10;j++) ran_array(x,KK+KK-1); /* warm things up */
00083   ran_arr_ptr=&ran_arr_started;
00084 }
