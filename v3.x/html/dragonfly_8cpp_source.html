00070       grp_output = dest_grp_ID;

00072     }
00073     else {
00074       grp_output = dest_grp_ID - 1;
00075       dest_grp_output = src_grp_ID;

00077     grp_output_RID = ((int) (grp_output / (gP))) + src_grp_ID * _grp_num_routers;

00081     dest_intm = grp_output_RID * gP;
00082 
00083     //hop count in source group
00084     if ((int)( src_intm / gP) == (int)( src / gP ) )
00085       src_hopcnt = 0;
00086     else
