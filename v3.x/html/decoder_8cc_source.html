00867                       drain_C_(w_L2_n[i], NCH, 1, 1, g_tp.cell_h_def, is_dram_);
00868         tf = rd * (c_intrinsic + c_load);
00869         this_delay = horowitz(inrisetime_nand2_path, tf, 0.5, 0.5, RISE);
00870         delay_nand2_path += this_delay;
00871         inrisetime_nand2_path = this_delay / (1.0 - 0.5);
00872         this_delay = horowitz(inrisetime_nand3_path, tf, 0.5, 0.5, RISE);
00873         delay_nand3_path += this_delay;
00874         inrisetime_nand3_path = this_delay / (1.0 - 0.5);
00875         power_L2.readOp.dynamic += (c_intrinsic + c_load) * Vdd * Vdd;
00876       }
00877 
00878       //Add delay of final inverter that drives the wordline decoders
00879       i = number_gates_L2 - 1;
00880       c_load = C_ld_predec_blk_out;
00881       rd = tr_R_on(w_L2_n[i], NCH, 1, is_dram_);
00882       c_intrinsic = drain_C_(w_L2_p[i], PCH, 1, 1, g_tp.cell_h_def, is_dram_) +
00883                     drain_C_(w_L2_n[i], NCH, 1, 1, g_tp.cell_h_def, is_dram_);
00884       tf = rd * (c_intrinsic + c_load) + R_wire_predec_blk_out * c_load / 2;
00885       this_delay = horowitz(inrisetime_nand2_path, tf, 0.5, 0.5, RISE);
00886       delay_nand2_path += this_delay;
00887       ret_val.first = this_delay / (1.0 - 0.5);
00888       this_delay = horowitz(inrisetime_nand3_path, tf, 0.5, 0.5, RISE);
00889       delay_nand3_path += this_delay;
00890       ret_val.second = this_delay / (1.0 - 0.5);
00891       power_L2.readOp.dynamic += (c_intrinsic + c_load) * Vdd * Vdd;
00892     }
00893   }
00894 
00895   delay = (ret_val.first > ret_val.second) ? ret_val.first : ret_val.second;
00896   return ret_val;

01259   ret_val.second = 0;  // outrisetime_nand3_path

01263 
01264   if (flag_driver_exists)

01266     for (i = 0; i < number_gates_nand2_path - 1; ++i)
01267     {
01268       rd = tr_R_on(width_nand2_path_n[i], NCH, 1, is_dram_);
01269       c_gate_load = gate_C(width_nand2_path_p[i+1] + width_nand2_path_n[i+1], 0.0, is_dram_);

01271                     drain_C_(width_nand2_path_n[i], NCH, 1, 1, g_tp.cell_h_def, is_dram_);
01272       tf = rd * (c_intrinsic + c_gate_load);

01274       delay_nand2_path += this_delay;
01275       inrisetime_nand2_path = this_delay / (1.0 - 0.5);
01276       power_nand2_path.readOp.dynamic += (c_gate_load + c_intrinsic) * 0.5 * Vdd * Vdd;
