01443     std::map<std::string, unsigned> cuobjdumpSectionMap;
01444     for (   std::list<cuobjdumpSection*>::iterator iter = cuobjdumpSectionList.begin();
01445             iter != cuobjdumpSectionList.end();

01449                 (capability <= forced_max_capability ||
01450                         forced_max_capability==0)) {
