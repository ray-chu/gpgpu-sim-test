00308 {
00309    const symbol *sym = symtab->lookup(name.c_str());
00310    fprintf(fp,"  %8s   ", name.c_str() );
00311    if( sym == NULL ) {
00312       fprintf(fp,"<unknown type> 0x%llx\n", (unsigned long long ) value.u64 );
00313       return;

00317       fprintf(fp,"<unknown type> 0x%llx\n", (unsigned long long ) value.u64 );

00321 
00322    switch ( ti.scalar_type() ) {
