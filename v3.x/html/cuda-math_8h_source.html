00272 int float2int(float a, enum cudaRoundMode mode)
00273 {
00274    int tmp;
00275    switch (mode) {
00276    case cuda_math::cudaRoundZero: tmp = truncf(a);     break;
00277    case cuda_math::cudaRoundNearest: tmp = nearbyintf(a); break;
00278    case cuda_math::cudaRoundMinInf: tmp = floorf(a);     break;
00279    case cuda_math::cudaRoundPosInf: tmp = ceilf(a);      break;
00280    default: abort();
00281    }
00282    return tmp;
