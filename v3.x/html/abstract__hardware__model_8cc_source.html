00343     unsigned segment_size = 0;
00344     unsigned warp_parts = m_config->mem_warp_parts;
00345     switch( data_size ) {
00346     case 1: segment_size = 32; break;
00347     case 2: segment_size = 64; break;
00348     case 4: case 8: case 16: segment_size = 128; break;
00349     }
00350     unsigned subwarp_size = m_config->warp_size / warp_parts;
00351 
00352     for( unsigned subwarp=0; subwarp <  warp_parts; subwarp++ ) {
00353         std::map<new_addr_type,transaction_info> subwarp_transactions;
00354 
00355         // step 1: find all transactions generated by this subwarp
00356         for( unsigned thread=subwarp*subwarp_size; thread<subwarp_size*(subwarp+1); thread++ ) {
00357             if( !active(thread) )
00358                 continue;
00359 
00360             unsigned data_size_coales = data_size;
00361             unsigned num_accesses = 1;
00362 
00363             if( space.get_type() == local_space || space.get_type() == param_space_local ) {
00364                // Local memory accesses >4B were split into 4B chunks
00365                if(data_size >= 4) {
00366                   data_size_coales = 4;
00367                   num_accesses = data_size/4;
00368                }
00369                // Otherwise keep the same data_size for sub-4B access to local memory
00370             }
00371 
00372 
00373             assert(num_accesses <= MAX_ACCESSES_PER_INSN_PER_THREAD);
00374 
00375             for(unsigned access=0; access<num_accesses; access++) {
00376                 new_addr_type addr = m_per_scalar_thread[thread].memreqaddr[access];
00377                 unsigned block_address = line_size_based_tag_func(addr,segment_size);
00378                 unsigned chunk = (addr&127)/32; // which 32-byte chunk within in a 128-byte chunk does this thread access?
00379                 transaction_info &info = subwarp_transactions[block_address];
00380 
00381                 // can only write to one segment
00382                 assert(block_address == line_size_based_tag_func(addr+data_size_coales-1,segment_size));

00385                 info.active.set(thread);
00386                 unsigned idx = (addr&127);

00388                     info.bytes.set(idx+i);

00390         }
00391

00497            // only upper 64 bytes used
00498            addr = addr+64;

00500            if(q[2]) h.set(0);
00501            if(q[3]) h.set(1);

00503            assert(lower_half_used && upper_half_used);
00504        }

00508            if(q[0]) h.set(0);

00522            addr = addr+32;

00635 void simt_stack::update( simt_mask_t &thread_done, addr_vector_t &next_pc, address_type recvg_pc, op_type next_inst_op )
00636 {
00637     assert(m_stack.size() > 0);
00638 
00639     assert( next_pc.size() == m_warp_size );
00640 
00641     simt_mask_t  top_active_mask = m_stack.back().m_active_mask;
00642     address_type top_recvg_pc = m_stack.back().m_recvg_pc;
00643     address_type top_pc = m_stack.back().m_pc; // the pc of the instruction just executed
00644     stack_entry_type top_type = m_stack.back().m_type;
00645 
00646     assert(top_active_mask.any());
00647 
00648     const address_type null_pc = -1;
00649     bool warp_diverged = false;
00650     address_type new_recvg_pc = null_pc;
00651     while (top_active_mask.any()) {
00652 
00653         // extract a group of threads with the same next PC among the active threads in the warp
00654         address_type tmp_next_pc = null_pc;
00655         simt_mask_t tmp_active_mask;
00656         for (int i = m_warp_size - 1; i >= 0; i--) {
00657             if ( top_active_mask.test(i) ) { // is this thread active?
00658                 if (thread_done.test(i)) {
00659                     top_active_mask.reset(i); // remove completed thread from active mask
00660                 } else if (tmp_next_pc == null_pc) {
00661                     tmp_next_pc = next_pc[i];
00662                     tmp_active_mask.set(i);
00663                     top_active_mask.reset(i);
00664                 } else if (tmp_next_pc == next_pc[i]) {
00665                     tmp_active_mask.set(i);
00666                     top_active_mask.reset(i);
00667                 }
00668             }
00669         }
00670 
00671         if(tmp_next_pc == null_pc) {
00672             assert(!top_active_mask.any()); // all threads done
00673             continue;
00674         }
00675 
00676         // HANDLE THE SPECIAL CASES FIRST
00677         if (next_inst_op== CALL_OPS)
00678         {
00679             // Since call is not a divergent instruction, all threads should have executed a call instruction
00680             assert(top_active_mask.any() == false);
00681 
00682             simt_stack_entry new_stack_entry;
00683             new_stack_entry.m_pc = tmp_next_pc;
00684             new_stack_entry.m_active_mask = tmp_active_mask;
00685             new_stack_entry.m_branch_div_cycle = gpu_sim_cycle+gpu_tot_sim_cycle;
00686             new_stack_entry.m_type = STACK_ENTRY_TYPE_CALL;
00687             m_stack.push_back(new_stack_entry);
00688             return;
00689 
00690         } else if(next_inst_op == RET_OPS && top_type==STACK_ENTRY_TYPE_CALL) {
00691             // pop the CALL Entry
00692             assert(top_active_mask.any() == false);
00693             m_stack.pop_back();
00694 
00695             assert(m_stack.size() > 0);
00696             m_stack.back().m_pc=tmp_next_pc;// set the PC of the stack top entry to return PC from  the call stack;
00697             // Check if the New top of the stack is reconverging
00698             if (tmp_next_pc == m_stack.back().m_recvg_pc && m_stack.back().m_type!=STACK_ENTRY_TYPE_CALL)
00699             {
00700                 assert(m_stack.back().m_type==STACK_ENTRY_TYPE_NORMAL);
00701                 m_stack.pop_back();
00702             }
00703             return;
00704         }
00705 
00706         // discard the new entry if its PC matches with reconvergence PC
00707         // that automatically reconverges the entry
00708         // If the top stack entry is CALL, dont reconverge.
00709         if (tmp_next_pc == top_recvg_pc && (top_type != STACK_ENTRY_TYPE_CALL)) continue;
00710 
00711         // this new entry is not converging
00712         // if this entry does not include thread from the warp, divergence occurs
00713         if (top_active_mask.any() && !warp_diverged ) {
00714             warp_diverged = true;
00715             // modify the existing top entry into a reconvergence entry in the pdom stack
00716             new_recvg_pc = recvg_pc;
00717             if (new_recvg_pc != top_recvg_pc) {
00718                 m_stack.back().m_pc = new_recvg_pc;
00719                 m_stack.back().m_branch_div_cycle = gpu_sim_cycle+gpu_tot_sim_cycle;
00720 
00721                 m_stack.push_back(simt_stack_entry());
00722             }
00723         }
00724 
00725         // discard the new entry if its PC matches with reconvergence PC
00726         if (warp_diverged && tmp_next_pc == new_recvg_pc) continue;
00727 
00728         // update the current top of pdom stack
00729         m_stack.back().m_pc = tmp_next_pc;
00730         m_stack.back().m_active_mask = tmp_active_mask;
00731         if (warp_diverged) {
00732             m_stack.back().m_calldepth = 0;
00733             m_stack.back().m_recvg_pc = new_recvg_pc;
00734         } else {
00735             m_stack.back().m_recvg_pc = top_recvg_pc;
00736         }
00737 
00738         m_stack.push_back(simt_stack_entry());
00739     }
00740     assert(m_stack.size() > 0);
00741     m_stack.pop_back();
00742 
00743 
00744     if (warp_diverged) {
00745         ptx_file_line_stats_add_warp_divergence(top_pc, 1); 
00746     }
