00237       // parse config file

00326 vector<string> tokenize_str(string const & data)
00327 {
00328   vector<string> values;
00329 
00330   // no elements, no braces --> empty list
00331   if(data.empty()) {
00332     return values;
00333   }
00334 
00335   // doesn't start with an opening brace --> treat as single element
00336   // note that this element can potentially contain nested lists 
00337   if(data[0] != '{') {
00338     values.push_back(data);
00339     return values;
00340   }
00341 
00342   size_t start = 1;
00343   int nested = 0;
00344 
00345   size_t curr = start;
00346 
00347   while(string::npos != (curr = data.find_first_of("{,}", curr))) {
00348     
00349     if(data[curr] == '{') {
00350       ++nested;
00351     } else if((data[curr] == '}') && nested) {
00352       --nested;
00353     } else if(!nested) {
00354       if(curr > start) {
00355     string token = data.substr(start, curr - start);
00356     values.push_back(token);
00357       }
00358       start = curr + 1;
00359     }
00360     ++curr;
00361   }
00362   assert(!nested);
00363 
00364   return values;
00365 }
