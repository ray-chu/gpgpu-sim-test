00120 // output statistics to a file
00121 void ptx_file_line_stats_write_file()
00122 {
00123     // check if stat collection is turned on
00124     if (enable_ptx_file_line_stats == 0) return;
00125 
00126     ptx_file_line_stats_map_t::iterator it;
00127     FILE * pfile;
00128 
00129     pfile = fopen(ptx_line_stats_filename, "w");
00130     fprintf(pfile,"kernel line : count latency dram_traffic smem_bk_conflicts smem_warp gmem_access_generated gmem_warp exposed_latency warp_divergence\n");
00131     for( it=ptx_file_line_stats_tracker.begin(); it != ptx_file_line_stats_tracker.end(); it++ ) {
00132         fprintf(pfile, "%s %i : ", it->first.st.c_str(), it->first.line);
00133         fprintf(pfile, "%lu ", it->second.exec_count);
00134         fprintf(pfile, "%llu ", it->second.latency);
00135         fprintf(pfile, "%llu ", it->second.dram_traffic);
00136         fprintf(pfile, "%llu ", it->second.smem_n_way_bank_conflict_total);
00137         fprintf(pfile, "%lu ", it->second.smem_warp_count);
00138         fprintf(pfile, "%llu ", it->second.gmem_n_access_total);
00139         fprintf(pfile, "%lu ", it->second.gmem_warp_count);
00140         fprintf(pfile, "%llu ", it->second.exposed_latency);
00141         fprintf(pfile, "%llu ", it->second.warp_divergence);
00142         fprintf(pfile, "\n");
00143     }
00144     fflush(pfile);
00145     fclose(pfile);

00197 class ptx_inflight_memory_insn_tracker
00198 {
00199 public:
00200     typedef std::map<const ptx_instruction *, int> insn_count_map;
00201 
00202     void add_count(const ptx_instruction * pInsn, int count = 1)
00203     {
00204         ptx_inflight_memory_insns[pInsn] += count;
00205     }
00206 
00207     void sub_count(const ptx_instruction * pInsn, int count = 1)
00208     {
00209         insn_count_map::iterator i_insncount; 
00210         i_insncount = ptx_inflight_memory_insns.find(pInsn);
00211 
00212         assert(i_insncount != ptx_inflight_memory_insns.end());
00213 
00214         i_insncount->second -= count;
00215 
00216         if (i_insncount->second <= 0) {
00217             ptx_inflight_memory_insns.erase(i_insncount);
00218         }
00219     }
00220 
00221     void attribute_exposed_latency(int count = 1)
00222     {
00223         insn_count_map &exlat_insnmap = ptx_inflight_memory_insns;
00224         insn_count_map::const_iterator i_exlatinsn;
00225 
00226         i_exlatinsn = exlat_insnmap.begin();
00227         for (; i_exlatinsn != exlat_insnmap.end(); ++i_exlatinsn) {
00228             const ptx_instruction *pInsn = i_exlatinsn->first;
00229             ptx_file_line_stats& line_stats = ptx_file_line_stats_tracker[ptx_file_line(pInsn->source_file(), pInsn->source_line())];
00230             line_stats.exposed_latency += count;
00231         }
00232     }
00233 
00234     insn_count_map ptx_inflight_memory_insns;
00235 };
