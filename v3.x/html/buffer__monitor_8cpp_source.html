00032 BufferMonitor::BufferMonitor( int inputs, int classes ) 
00033 : _cycles(0), _inputs(inputs), _classes(classes) {
00034   _reads.resize(inputs * classes, 0) ;

00038 int BufferMonitor::index( int input, int cl ) const {
00039   assert((input >= 0) && (input < _inputs)); 
00040   assert((cl >= 0) && (cl < _classes));
00041   return cl + _classes * input ;
00042 }
