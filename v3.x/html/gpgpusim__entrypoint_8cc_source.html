00067 void *gpgpu_sim_thread_sequential(void*)
00068 {
00069    // at most one kernel running at a time
00070    bool done;
00071    do {
00072       sem_wait(&g_sim_signal_start);
00073       done = true;
00074       if( g_the_gpu->get_more_cta_left() ) {
00075           done = false;
00076           g_the_gpu->init();
00077           while( g_the_gpu->active() ) {
00078               g_the_gpu->cycle();
00079               g_the_gpu->deadlock_check();
00080           }
00081           g_the_gpu->print_stats();
00082           g_the_gpu->update_stats();
00083           print_simulation_time();
00084       }
00085       sem_post(&g_sim_signal_finish);
00086    } while(!done);
00087    sem_post(&g_sim_signal_exit);
00088    return NULL;
00089 }

00095 void *gpgpu_sim_thread_concurrent(void*)
00096 {
00097     // concurrent kernel execution simulation thread
00098     do {
00099        if(g_debug_execution >= 3) {
00100           printf("GPGPU-Sim: *** simulation thread starting and spinning waiting for work ***\n");
00101           fflush(stdout);
00102        }
00103         while( g_stream_manager->empty_protected() && !g_sim_done )
00104             ;
00105         if(g_debug_execution >= 3) {
00106            printf("GPGPU-Sim: ** START simulation thread (detected work) **\n");
00107            g_stream_manager->print(stdout);
00108            fflush(stdout);
00109         }
00110         pthread_mutex_lock(&g_sim_lock);
00111         g_sim_active = true;
00112         pthread_mutex_unlock(&g_sim_lock);
00113         bool active = false;
00114         bool sim_cycles = false;
00115         g_the_gpu->init();
00116         do {
00117             // check if a kernel has completed
00118             // launch operation on device if one is pending and can be run
00119 
00120             // Need to break this loop when a kernel completes. This was a
00121             // source of non-deterministic behaviour in GPGPU-Sim (bug 147).
00122             // If another stream operation is available, g_the_gpu remains active,
00123             // causing this loop to not break. If the next operation happens to be
00124             // another kernel, the gpu is not re-initialized and the inter-kernel
00125             // behaviour may be incorrect. Check that a kernel has finished and
00126             // no other kernel is currently running.
00127             if(g_stream_manager->operation(&sim_cycles) && !g_the_gpu->active())
00128                 break;
00129 
00130             if( g_the_gpu->active() ) {
00131                 g_the_gpu->cycle();
00132                 sim_cycles = true;
00133                 g_the_gpu->deadlock_check();
00134             }
00135             active=g_the_gpu->active() || !g_stream_manager->empty_protected();
00136         } while( active );
00137         if(g_debug_execution >= 3) {
00138            printf("GPGPU-Sim: ** STOP simulation thread (no work) **\n");
00139            fflush(stdout);
00140         }
00141         if(sim_cycles) {
00142             g_the_gpu->update_stats();
00143             print_simulation_time();
00144         }
00145         pthread_mutex_lock(&g_sim_lock);
00146         g_sim_active = false;
00147         pthread_mutex_unlock(&g_sim_lock);
00148     } while( !g_sim_done );
00149     if(g_debug_execution >= 3) {
00150        printf("GPGPU-Sim: *** simulation thread exiting ***\n");
00151        fflush(stdout);
00152     }
00153     sem_post(&g_sim_signal_exit);
00154     return NULL;
00155 }
