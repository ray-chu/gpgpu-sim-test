00037 double wire_resistance(double resistivity, double wire_width, double wire_thickness,
00038     double barrier_thickness, double dishing_thickness, double alpha_scatter)
00039 {
00040   double resistance;
00041   resistance = alpha_scatter * resistivity /((wire_thickness - barrier_thickness - dishing_thickness)*(wire_width - 2 * barrier_thickness));
00042   return(resistance);
00043 }

00045 double wire_capacitance(double wire_width, double wire_thickness, double wire_spacing,
00046     double ild_thickness, double miller_value, double horiz_dielectric_constant,
00047     double vert_dielectric_constant, double fringe_cap)
00048 {
00049   double vertical_cap, sidewall_cap, total_cap;
00050   vertical_cap = 2 * PERMITTIVITY_FREE_SPACE * vert_dielectric_constant * wire_width / ild_thickness;
00051   sidewall_cap = 2 * PERMITTIVITY_FREE_SPACE * miller_value * horiz_dielectric_constant * wire_thickness / wire_spacing;
00052   total_cap = vertical_cap + sidewall_cap + fringe_cap;
00053   return(total_cap);
