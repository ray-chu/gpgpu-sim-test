00060 double Component::compute_diffusion_width(int num_stacked_in, int num_folded_tr)
00061 {
00062   double w_poly = g_ip->F_sz_um;
00063   double spacing_poly_to_poly = g_tp.w_poly_contact + 2 * g_tp.spacing_poly_to_contact;
00064   double total_diff_w = 2 * spacing_poly_to_poly +  // for both source and drain
00065                         num_stacked_in * w_poly +
00066                         (num_stacked_in - 1) * g_tp.spacing_poly_to_poly;
00067 
00068   if (num_folded_tr > 1)
00069   {
00070     total_diff_w += (num_folded_tr - 2) * 2 * spacing_poly_to_poly +
00071                     (num_folded_tr - 1) * num_stacked_in * w_poly +
00072                     (num_folded_tr - 1) * (num_stacked_in - 1) * g_tp.spacing_poly_to_poly;
00073   }
00074 
00075   return total_diff_w;
00076 }

00080 double Component::compute_gate_area(
00081     int gate_type,
00082     int num_inputs,
00083     double w_pmos,
00084     double w_nmos,
00085     double h_gate)
00086 {
00087   if (w_pmos <= 0.0 || w_nmos <= 0.0)
00088   {
00089     return 0.0;
00090   }
00091 
00092   double w_folded_pmos, w_folded_nmos;
00093   int    num_folded_pmos, num_folded_nmos;
00094   double total_ndiff_w, total_pdiff_w;
00095   Area gate;
00096 
00097   double h_tr_region  = h_gate - 2 * g_tp.HPOWERRAIL;
00098   double ratio_p_to_n = w_pmos / (w_pmos + w_nmos);
00099 
00100   if (ratio_p_to_n >= 1 || ratio_p_to_n <= 0)
00101   {
00102     return 0.0;
00103   }
00104 
00105   w_folded_pmos  = (h_tr_region - g_tp.MIN_GAP_BET_P_AND_N_DIFFS) * ratio_p_to_n;
00106   w_folded_nmos  = (h_tr_region - g_tp.MIN_GAP_BET_P_AND_N_DIFFS) * (1 - ratio_p_to_n);
00107   assert(w_folded_pmos > 0);
00108 
00109   num_folded_pmos = (int) (ceil(w_pmos / w_folded_pmos));
00110   num_folded_nmos = (int) (ceil(w_nmos / w_folded_nmos));
00111 
00112   switch (gate_type)
00113   {
00114     case INV:
00115       total_ndiff_w = compute_diffusion_width(1, num_folded_nmos);
00116       total_pdiff_w = compute_diffusion_width(1, num_folded_pmos);
00117       break;
00118 
00119     case NOR:
00120       total_ndiff_w = compute_diffusion_width(1, num_inputs * num_folded_nmos);
00121       total_pdiff_w = compute_diffusion_width(num_inputs, num_folded_pmos);
00122       break;
00123 
00124     case NAND:
00125       total_ndiff_w = compute_diffusion_width(num_inputs, num_folded_nmos);
00126       total_pdiff_w = compute_diffusion_width(1, num_inputs * num_folded_pmos);
00127       break;
00128     default:
00129       cout << "Unknown gate type: " << gate_type << endl;
00130       exit(1);
00131   }
00132 
00133   gate.w = MAX(total_ndiff_w, total_pdiff_w);
00134 
00135   if (w_folded_nmos > w_nmos)
00136   {
00137     //means that the height of the gate can
00138     //be made smaller than the input height specified, so calculate the height of the gate.
00139     gate.h = w_nmos + w_pmos + g_tp.MIN_GAP_BET_P_AND_N_DIFFS + 2 * g_tp.HPOWERRAIL;
00140   }
00141   else
00142   {
00143     gate.h = h_gate;
00144   }
00145   return gate.get_area();
00146 }

00150 double Component::compute_tr_width_after_folding(
00151     double input_width,
00152     double threshold_folding_width)
00153 {//This is actually the width of the cell not the width of a device.
00154 //The width of a cell and the width of a device is orthogonal.
00155   if (input_width <= 0)
00156   {
00157     return 0;
00158   }
00159 
00160   int    num_folded_tr        = (int) (ceil(input_width / threshold_folding_width));
00161   double spacing_poly_to_poly = g_tp.w_poly_contact + 2 * g_tp.spacing_poly_to_contact;
00162   double width_poly           = g_ip->F_sz_um;
00163   double total_diff_width     = num_folded_tr * width_poly + (num_folded_tr + 1) * spacing_poly_to_poly;
00164 
00165   return total_diff_width;
00166 }

00170 double Component::height_sense_amplifier(double pitch_sense_amp)
00171 {
00172   // compute the height occupied by all PMOS transistors
00173   double h_pmos_tr = compute_tr_width_after_folding(g_tp.w_sense_p, pitch_sense_amp) * 2 +
00174                      compute_tr_width_after_folding(g_tp.w_iso, pitch_sense_amp) +
00175                      2 * g_tp.MIN_GAP_BET_SAME_TYPE_DIFFS;
00176 
00177   // compute the height occupied by all NMOS transistors
00178   double h_nmos_tr = compute_tr_width_after_folding(g_tp.w_sense_n, pitch_sense_amp) * 2 +
00179                      compute_tr_width_after_folding(g_tp.w_sense_en, pitch_sense_amp) +
00180                      2 * g_tp.MIN_GAP_BET_SAME_TYPE_DIFFS;
00181 
00182   // compute total height by considering gap between the p and n diffusion areas
00183   return h_pmos_tr + h_nmos_tr + g_tp.MIN_GAP_BET_P_AND_N_DIFFS;
00184 }

00188 int Component::logical_effort(
00189     int num_gates_min,
00190     double g,
00191     double F,
00192     double * w_n,
00193     double * w_p,
00194     double C_load,
00195     double p_to_n_sz_ratio,
00196     bool   is_dram_,
00197     bool   is_wl_tr_,
00198     double max_w_nmos)

00201 
00202   // check if num_gates is odd. if so, add 1 to make it even
00203   num_gates+= (num_gates % 2) ? 1 : 0;
00204   num_gates = MAX(num_gates, num_gates_min);
00205 
00206   // recalculate the effective fanout of each stage
00207   double f = pow(F, 1.0 / num_gates);
00208   int    i = num_gates - 1;
00209   double C_in = C_load / f;
00210   w_n[i]  = (1.0 / (1.0 + p_to_n_sz_ratio)) * C_in / gate_C(1, 0, is_dram_, false, is_wl_tr_);
00211   w_n[i]  = MAX(w_n[i], g_tp.min_w_nmos_);
00212   w_p[i]  = p_to_n_sz_ratio * w_n[i];
00213 
00214   if (w_n[i] > max_w_nmos)
00215   {
00216     double C_ld = gate_C((1 + p_to_n_sz_ratio) * max_w_nmos, 0, is_dram_, false, is_wl_tr_);
00217     F = g * C_ld / gate_C(w_n[0] + w_p[0], 0, is_dram_, false, is_wl_tr_);
00218     num_gates = (int) (log(F) / log(fopt)) + 1;
00219     num_gates+= (num_gates % 2) ? 1 : 0;
00220     num_gates = MAX(num_gates, num_gates_min);
00221     f = pow(F, 1.0 / (num_gates - 1));
00222     i = num_gates - 1;
00223     w_n[i]  = max_w_nmos;
00224     w_p[i]  = p_to_n_sz_ratio * w_n[i];
00225   }
00226 
00227   for (i = num_gates - 2; i >= 1; i--)
00228   {
00229     w_n[i] = MAX(w_n[i+1] / f, g_tp.min_w_nmos_);
00230     w_p[i] = p_to_n_sz_ratio * w_n[i];
00231   }
00232 
00233   assert(num_gates <= MAX_NUMBER_GATES_STAGE);
00234   return num_gates;
00235 }
