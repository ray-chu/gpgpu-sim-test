00080 void VC::AddFlit( Flit *f )
00081 {
00082   assert(f);
00083 
00084   if(_expected_pid >= 0) {
00085     if(f->pid != _expected_pid) {
00086       ostringstream err;
00087       err << "Received flit " << f->id << " with unexpected packet ID: " << f->pid 
00088       << " (expected: " << _expected_pid << ")";
00089       Error(err.str());
00090     } else if(f->tail) {
00091       _expected_pid = -1;
00092     }
00093   } else if(!f->tail) {
00094     _expected_pid = f->pid;
00095   }
00096     
00097   // update flit priority before adding to VC buffer
00098   if(_pri_type == local_age_based) {
00099     f->pri = numeric_limits<int>::max() - GetSimTime();
00100     assert(f->pri >= 0);
00101   } else if(_pri_type == hop_count_based) {
00102     f->pri = f->hops;
00103     assert(f->pri >= 0);
00104   }
00105 
00106   _buffer.push_back(f);
00107   UpdatePriority();
00108 }

00157 void VC::UpdatePriority()
00158 {
00159   if(_buffer.empty()) return;
00160   if(_pri_type == queue_length_based) {
00161     _pri = _buffer.size();
00162   } else if(_pri_type != none) {
00163     Flit * f = _buffer.front();
00164     if((_pri_type != local_age_based) && _priority_donation) {
00165       Flit * df = f;
00166       for(size_t i = 1; i < _buffer.size(); ++i) {
00167     Flit * bf = _buffer[i];
00168     if(bf->pri > df->pri) df = bf;
00169       }
00170       if((df != f) && (df->watch || f->watch)) {
00171     *gWatchOut << GetSimTime() << " | " << FullName() << " | "
00172             << "Flit " << df->id
00173             << " donates priority to flit " << f->id
00174             << "." << endl;
00175       }
00176       f = df;
00177     }
00178     if(f->watch)
00179       *gWatchOut << GetSimTime() << " | " << FullName() << " | "
00180           << "Flit " << f->id
00181           << " sets priority to " << f->pri
00182           << "." << endl;
00183     _pri = f->pri;
00184   }
00185 }
