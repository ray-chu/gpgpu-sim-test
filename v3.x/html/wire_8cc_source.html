00217  * The fall time of an input signal to the first stage of a circuit is
00218  * assumed to be same as the fall time of the output signal of two
00219  * inverters connected in series (refer: CACTI 1 Technical report,
00220  * section 6.1.3)
00221  */
00222   double
00223 Wire::signal_fall_time ()
00224 {
00225 
00226   /* rise time of inverter 1's output */
00227   double rt;
00228   /* fall time of inverter 2's output */
00229   double ft;
00230   double timeconst;
00231 
00232   timeconst = (drain_C_(g_tp.min_w_nmos_, NCH, 1, 1, g_tp.cell_h_def) +
00233       drain_C_(min_w_pmos, PCH, 1, 1, g_tp.cell_h_def) +
00234       gate_C(min_w_pmos + g_tp.min_w_nmos_, 0)) *
00235     tr_R_on(min_w_pmos, PCH, 1);
00236   rt = horowitz (0, timeconst, deviceType->Vth/deviceType->Vdd, deviceType->Vth/deviceType->Vdd, FALL) / (deviceType->Vdd - deviceType->Vth);
00237   timeconst = (drain_C_(g_tp.min_w_nmos_, NCH, 1, 1, g_tp.cell_h_def) +
00238       drain_C_(min_w_pmos, PCH, 1, 1, g_tp.cell_h_def) +
00239       gate_C(min_w_pmos + g_tp.min_w_nmos_, 0)) *
00240     tr_R_on(g_tp.min_w_nmos_, NCH, 1);
00241   ft = horowitz (rt, timeconst, deviceType->Vth/deviceType->Vdd, deviceType->Vth/deviceType->Vdd, RISE) / deviceType->Vth;
00242   return ft;

00392 /*
00393  * Calculates the delay, power and area of the transmitter circuit.
00394  *
00395  * The transmitter delay is the sum of nand gate delay, inverter delay
00396  * low swing nmos delay, and the wire delay
00397  * (ref: Technical report 6)
00398  */
00399   void
00400 Wire::low_swing_model()
00401 {
00402   double len = wire_length;
00403   double beta = pmos_to_nmos_sz_ratio();
00404 
00405 
00406   double inputrise = (in_rise_time == 0) ? signal_rise_time() : in_rise_time;
00407 
00408   /* Final nmos low swing driver size calculation:
00409    * Try to size the driver such that the delay
00410    * is less than 8FO4.
00411    * If the driver size is greater than
00412    * the max allowable size, assume max size for the driver.
00413    * In either case, recalculate the delay using
00414    * the final driver size assuming slow input with
00415    * finite rise time instead of ideal step input
00416    *
00417    * (ref: Technical report 6)
00418    */
00419   double cwire = wire_cap(len); /* load capacitance */
00420   double rwire = wire_res(len);
00421 
00422 #define RES_ADJ (8.6) // Increase in resistance due to low driving vol.
00423 
00424   double driver_res = (-8*g_tp.FO4/(log(0.5) * cwire))/RES_ADJ;
00425   double nsize = R_to_w(driver_res, NCH);
00426 
00427   nsize = MIN(nsize, g_tp.max_w_nmos_);
00428   nsize = MAX(nsize, g_tp.min_w_nmos_);
00429 
00430   if(rwire*cwire > 8*g_tp.FO4)
00431   {
00432     nsize = g_tp.max_w_nmos_;
00433   }
00434 
00435   // size the inverter appropriately to minimize the transmitter delay
00436   // Note - In order to minimize leakage, we are not adding a set of inverters to
00437   // bring down delay. Instead, we are sizing the single gate
00438   // based on the logical effort.
00439   double st_eff   = sqrt((2+beta/1+beta)*gate_C(nsize, 0)/(gate_C(2*g_tp.min_w_nmos_, 0)
00440         + gate_C(2*min_w_pmos, 0)));
00441   double req_cin  = ((2+beta/1+beta)*gate_C(nsize, 0))/st_eff;
00442   double inv_size = req_cin/(gate_C(min_w_pmos, 0) + gate_C(g_tp.min_w_nmos_, 0));
00443   inv_size = MAX(inv_size, 1);
00444 
00445   /* nand gate delay */
00446   double res_eq = (2 * tr_R_on(g_tp.min_w_nmos_, NCH, 1));
00447   double cap_eq = 2 * drain_C_(min_w_pmos, PCH, 1, 1, g_tp.cell_h_def) +
00448     drain_C_(2*g_tp.min_w_nmos_, NCH, 1, 1, g_tp.cell_h_def) +
00449     gate_C(inv_size*g_tp.min_w_nmos_, 0) +
00450     gate_C(inv_size*min_w_pmos, 0);
00451 
00452   double timeconst = res_eq * cap_eq;
00453 
00454   delay = horowitz(inputrise, timeconst, deviceType->Vth/deviceType->Vdd,
00455       deviceType->Vth/deviceType->Vdd, RISE);
00456   double temp_power = cap_eq*deviceType->Vdd*deviceType->Vdd;
00457 
00458   inputrise = delay / (deviceType->Vdd - deviceType->Vth); /* for the next stage */
00459 
00460   /* Inverter delay:
00461    * The load capacitance of this inv depends on
00462    * the gate capacitance of the final stage nmos
00463    * transistor which in turn depends on nsize
00464    */
00465   res_eq = tr_R_on(inv_size*min_w_pmos, PCH, 1);
00466   cap_eq = drain_C_(inv_size*min_w_pmos, PCH, 1, 1, g_tp.cell_h_def) +
00467     drain_C_(inv_size*g_tp.min_w_nmos_, NCH, 1, 1, g_tp.cell_h_def) +
00468     gate_C(nsize, 0);
00469   timeconst = res_eq * cap_eq;
00470 
00471   delay += horowitz(inputrise, timeconst, deviceType->Vth/deviceType->Vdd,
00472       deviceType->Vth/deviceType->Vdd, FALL);
00473   temp_power += cap_eq*deviceType->Vdd*deviceType->Vdd;
00474 
00475 
00476   transmitter.delay = delay;
00477   transmitter.power.readOp.dynamic = temp_power*2; /* since it is a diff. model*/
00478   transmitter.power.readOp.leakage = deviceType->Vdd *
00479     (4 * cmos_Isub_leakage(g_tp.min_w_nmos_, min_w_pmos, 2, nand) +
00480      4 * cmos_Isub_leakage(g_tp.min_w_nmos_, min_w_pmos, 1, inv));
00481 
00482   transmitter.power.readOp.gate_leakage = deviceType->Vdd *
00483     (4 * cmos_Ig_leakage(g_tp.min_w_nmos_, min_w_pmos, 2, nand) +
00484      4 * cmos_Ig_leakage(g_tp.min_w_nmos_, min_w_pmos, 1, inv));
00485 
00486   inputrise = delay / deviceType->Vth;
00487 
00488   /* nmos delay + wire delay */
00489   cap_eq = cwire + drain_C_(nsize, NCH, 1, 1, g_tp.cell_h_def)*2 +
00490     nsense * sense_amp_input_cap(); //+receiver cap
00491   /*
00492    * NOTE: nmos is used as both pull up and pull down transistor
00493    * in the transmitter. This is because for low voltage swing, drive
00494    * resistance of nmos is less than pmos
00495    * (for a detailed graph ref: On-Chip Wires: Scaling and Efficiency)
00496    */
00497   timeconst = (tr_R_on(nsize, NCH, 1)*RES_ADJ) * (cwire +
00498       drain_C_(nsize, NCH, 1, 1, g_tp.cell_h_def)*2) +
00499     rwire*cwire/2 +
00500     (tr_R_on(nsize, NCH, 1)*RES_ADJ + rwire) *
00501     nsense * sense_amp_input_cap();
00502 
00503   /*
00504    * since we are pre-equalizing and overdriving the low
00505    * swing wires, the net time constant is less

00551 void Wire::delay_optimal_wire ()
00552 {
00553   double len       = wire_length;
00554   //double min_wire_width = wire_width; //m
00555   double beta = pmos_to_nmos_sz_ratio();
00556   double switching = 0;  // switching energy
00557   double short_ckt = 0;  // short-circuit energy
00558   double tc        = 0;  // time constant
00559   // input cap of min sized driver
00560   double input_cap = gate_C(g_tp.min_w_nmos_ + min_w_pmos, 0);
00561 
00562    // output parasitic capacitance of
00563    // the min. sized driver
00564   double out_cap = drain_C_(min_w_pmos, PCH, 1, 1, g_tp.cell_h_def) +
00565     drain_C_(g_tp.min_w_nmos_, NCH, 1, 1, g_tp.cell_h_def);
00566   // drive resistance
00567   double out_res = (tr_R_on(g_tp.min_w_nmos_, NCH, 1) +
00568       tr_R_on(min_w_pmos, PCH, 1))/2;
00569   double wr = wire_res(len); //ohm
00570 
00571   // wire cap /m
00572   double wc = wire_cap(len);
00573 
00574   // size the repeater such that the delay of the wire is minimum
00575   double repeater_scaling = sqrt(out_res*wc/(wr*input_cap)); // len will cancel
00576 
00577    // calc the optimum spacing between the repeaters (m)
00578 
00579   repeater_spacing = sqrt(2 * out_res * (out_cap + input_cap)/
00580       ((wr/len)*(wc/len)));
00581   repeater_size = repeater_scaling;
00582 
00583   switching = (repeater_scaling * (input_cap + out_cap) +
00584       repeater_spacing * (wc/len)) * deviceType->Vdd * deviceType->Vdd;
00585 
00586   tc = out_res * (input_cap + out_cap) +
00587     out_res * wc/len * repeater_spacing/repeater_scaling +
00588     wr/len * repeater_spacing * input_cap * repeater_scaling +
00589     0.5 * (wr/len) * (wc/len)* repeater_spacing * repeater_spacing;
00590 
00591   delay = 0.693 * tc * len/repeater_spacing;
00592 
00593 #define Ishort_ckt 65e-6 /* across all tech Ref:Banerjee et al. {IEEE TED} */
00594   short_ckt = deviceType->Vdd * g_tp.min_w_nmos_ * Ishort_ckt * 1.0986 *
00595     repeater_scaling * tc;
00596 
00597   area.set_area((len/repeater_spacing) *
00598                 compute_gate_area(INV, 1, min_w_pmos * repeater_scaling,
00599                                           g_tp.min_w_nmos_ * repeater_scaling, g_tp.cell_h_def));
00600   power.readOp.dynamic = ((len/repeater_spacing)*(switching + short_ckt));
00601   power.readOp.leakage = ((len/repeater_spacing)*
00602       deviceType->Vdd*
00603       cmos_Isub_leakage(g_tp.min_w_nmos_*repeater_scaling, beta*g_tp.min_w_nmos_*repeater_scaling, 1, inv));
00604   power.readOp.gate_leakage = ((len/repeater_spacing)*
00605       deviceType->Vdd*
00606       cmos_Ig_leakage(g_tp.min_w_nmos_*repeater_scaling, beta*g_tp.min_w_nmos_*repeater_scaling, 1, inv));
00607 }
