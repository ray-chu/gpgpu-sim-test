00287 void function_info::create_basic_blocks()
00288 {
00289    std::list<ptx_instruction*> leaders;
00290    std::list<ptx_instruction*>::iterator i, l;
00291 
00292    // first instruction is a leader
00293    i=m_instructions.begin();
00294    leaders.push_back(*i);
00295    i++;
00296    while( i!=m_instructions.end() ) {
00297       ptx_instruction *pI = *i;
00298       if( pI->is_label() ) {
00299          leaders.push_back(pI);
00300          i = find_next_real_instruction(++i);
00301       } else {
00302          switch( pI->get_opcode() ) {
00303          case BRA_OP: case RET_OP: case EXIT_OP: case RETP_OP: case BREAK_OP: 
00304             i++;
00305             if( i != m_instructions.end() ) 
00306                leaders.push_back(*i);
00307             i = find_next_real_instruction(i);
00308             break;
00309          case CALL_OP: case CALLP_OP:
00310             if( pI->has_pred() ) {
00311                printf("GPGPU-Sim PTX: Warning found predicated call\n");
00312                i++;
00313                if( i != m_instructions.end() ) 
00314                   leaders.push_back(*i);
00315                i = find_next_real_instruction(i);
00316             } else i++;
00317             break;
00318          default:
00319             i++;
00320          }
00321       } 
00322    }
00323 
00324    if( leaders.empty() ) {
00325       printf("GPGPU-Sim PTX: Function \'%s\' has no basic blocks\n", m_name.c_str());
00326       return;
00327    }
00328 
00329    unsigned bb_id = 0;
00330    l=leaders.begin();
00331    i=m_instructions.begin();
00332    m_basic_blocks.push_back( new basic_block_t(bb_id++,*find_next_real_instruction(i),NULL,1,0) );
00333    ptx_instruction *last_real_inst=*(l++);
00334 
00335    for( ; i!=m_instructions.end(); i++ ) {
00336       ptx_instruction *pI = *i;
00337       if( l != leaders.end() && *i == *l ) {
00338          // found start of next basic block
00339          m_basic_blocks.back()->ptx_end = last_real_inst;
00340          if( find_next_real_instruction(i) != m_instructions.end() ) { // if not bogus trailing label
00341             m_basic_blocks.push_back( new basic_block_t(bb_id++,*find_next_real_instruction(i),NULL,0,0) );
00342             last_real_inst = *find_next_real_instruction(i);
00343          }
00344          // start search for next leader
00345          l++;
00346       }
00347       pI->assign_bb( m_basic_blocks.back() );
00348       if( !pI->is_label() ) last_real_inst = pI;
00349    }
00350    m_basic_blocks.back()->ptx_end = last_real_inst;
00351    m_basic_blocks.push_back( /*exit basic block*/ new basic_block_t(bb_id,NULL,NULL,0,1) );
00352 }

00538 void intersect( std::set<int> &A, const std::set<int> &B )
00539 {
00540    // return intersection of A and B in A
00541    for( std::set<int>::iterator a=A.begin(); a!=A.end(); ) {    
00542       std::set<int>::iterator a_next = a;
00543       a_next++;
00544       if( B.find(*a) == B.end() ) {
00545          A.erase(*a);
00546          a = a_next;
00547       } else 
00548          a++;
00549    }

00571 void function_info::find_dominators( )
00572 {  
00573    // find dominators using algorithm of Muchnick's Adv. Compiler Design & Implemmntation Fig 7.14 
00574    printf("GPGPU-Sim PTX: Finding dominators for \'%s\'...\n", m_name.c_str() );
00575    fflush(stdout);
00576    assert( m_basic_blocks.size() >= 2 ); // must have a distinquished entry block
00577    std::vector<basic_block_t*>::iterator bb_itr = m_basic_blocks.begin();
00578    (*bb_itr)->dominator_ids.insert((*bb_itr)->bb_id);  // the only dominator of the entry block is the entry
00579    //copy all basic blocks to all dominator lists EXCEPT for the entry block
00580    for (++bb_itr;bb_itr != m_basic_blocks.end(); bb_itr++) { 
00581       for (unsigned i = 0; i < m_basic_blocks.size(); i++) 
00582          (*bb_itr)->dominator_ids.insert(i);
00583    }
00584    bool change = true;
00585    while (change) {
00586       change = false;
00587       for ( int h = 1/*skip entry*/; h < m_basic_blocks.size(); ++h ) {
00588          assert( m_basic_blocks[h]->bb_id == (unsigned)h );
00589          std::set<int> T;
00590          for (unsigned i=0;i< m_basic_blocks.size();i++) 
00591             T.insert(i);
00592          for ( std::set<int>::iterator s = m_basic_blocks[h]->predecessor_ids.begin();s != m_basic_blocks[h]->predecessor_ids.end();s++) 
00593             intersect(T, m_basic_blocks[*s]->dominator_ids);
00594          T.insert(h);
00595          if (!is_equal(T, m_basic_blocks[h]->dominator_ids)) {
00596             change = true;
00597             m_basic_blocks[h]->dominator_ids = T;
00598          }
00599       }
00600    }
00601    //clean the basic block of dominators of it has no predecessors -- except for entry block
00602    bb_itr = m_basic_blocks.begin();
00603    for (++bb_itr;bb_itr != m_basic_blocks.end(); bb_itr++) {
00604       if ((*bb_itr)->predecessor_ids.empty())
00605          (*bb_itr)->dominator_ids.clear();
00606    }

00640 void function_info::find_ipostdominators( )
00641 {  
00642    // find immediate postdominator blocks, using algorithm of
00643    // Muchnick's Adv. Compiler Design & Implemmntation Fig 7.15 
00644    printf("GPGPU-Sim PTX: Finding immediate postdominators for \'%s\'...\n", m_name.c_str() );
00645    fflush(stdout);
00646    assert( m_basic_blocks.size() >= 2 ); // must have a distinquished exit block
00647    for (unsigned i=0; i<m_basic_blocks.size(); i++) { //initialize Tmp(n) to all pdoms of n except for n
00648       m_basic_blocks[i]->Tmp_ids = m_basic_blocks[i]->postdominator_ids;
00649       assert( m_basic_blocks[i]->bb_id == i );
00650       m_basic_blocks[i]->Tmp_ids.erase(i);
00651    }
00652    for ( int n = m_basic_blocks.size()-2; n >=0;--n) {
00653       // point iterator to basic block before the exit
00654       for( std::set<int>::iterator s=m_basic_blocks[n]->Tmp_ids.begin(); s != m_basic_blocks[n]->Tmp_ids.end(); s++ ) {
00655          int bb_s = *s;
00656          for( std::set<int>::iterator t=m_basic_blocks[n]->Tmp_ids.begin(); t != m_basic_blocks[n]->Tmp_ids.end(); ) {
00657             std::set<int>::iterator t_next = t; t_next++; // might erase thing pointed to be t, invalidating iterator t
00658             if( *s == *t ) {
00659                t = t_next;
00660                continue;
00661             }
00662             int bb_t = *t;
00663             if( m_basic_blocks[bb_s]->postdominator_ids.find(bb_t) != m_basic_blocks[bb_s]->postdominator_ids.end() ) 
00664                 m_basic_blocks[n]->Tmp_ids.erase(bb_t);
00665             t = t_next;
00666          }
00667       }
00668    }
00669    unsigned num_ipdoms=0;
00670    for ( int n = m_basic_blocks.size()-1; n >=0;--n) {
00671       assert( m_basic_blocks[n]->Tmp_ids.size() <= 1 ); 
00672          // if the above assert fails we have an error in either postdominator 
00673          // computation, the flow graph does not have a unique exit, or some other error
00674       if( !m_basic_blocks[n]->Tmp_ids.empty() ) {
00675          m_basic_blocks[n]->immediatepostdominator_id = *m_basic_blocks[n]->Tmp_ids.begin();
00676          num_ipdoms++;
00677       }
00678    }
00679    assert( num_ipdoms == m_basic_blocks.size()-1 ); 
00680       // the exit node does not have an immediate post dominator, but everyone else should
00681 }

00953 static std::list<operand_info> check_operands( int opcode,
00954                                         const std::list<int> &scalar_type,
00955                                         const std::list<operand_info> &operands )
00956 {
00957    static int g_warn_literal_operands_two_type_inst;
00958     if( (opcode == CVT_OP) || (opcode == SET_OP) || (opcode == SLCT_OP) || (opcode == TEX_OP) ) {
00959         // just make sure these do not have have const operands... 
00960         if( !g_warn_literal_operands_two_type_inst ) {
00961             std::list<operand_info>::const_iterator o;
00962             for( o = operands.begin(); o != operands.end(); o++ ) {
00963                 const operand_info &op = *o;
00964                 if( op.is_literal() ) {
00965                     printf("GPGPU-Sim PTX: PTX uses two scalar type intruction with literal operand.\n");
00966                     g_warn_literal_operands_two_type_inst = 1;
00967                 }
00968             }
00969         }
00970     } else {
00971         assert( scalar_type.size() < 2 );
00972         if( scalar_type.size() == 1 ) {
00973             std::list<operand_info> result;
00974             int inst_type = scalar_type.front();
00975             std::list<operand_info>::const_iterator o;
00976             for( o = operands.begin(); o != operands.end(); o++ ) {
00977                 const operand_info &op = *o;
00978                 if( op.is_literal() ) {
00979                     if( (op.get_type() == double_op_t) && (inst_type == F32_TYPE) ) {
00980                         ptx_reg_t v = op.get_literal_value();
00981                         float u = (float)v.f64;
00982                         operand_info n(u);
00983                         result.push_back(n);
00984                     } else {
00985                         result.push_back(op);
00986                     }
00987                 } else {
00988                         result.push_back(op);
00989                 }
00990             }
00991             return result;
00992         } 
00993     }
00994     return operands;
00995 }
