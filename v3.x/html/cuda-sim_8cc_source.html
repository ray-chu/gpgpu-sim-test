00787 {
00788    pc = m_PC;
00789    isize = m_inst_size;
00790    for( unsigned i=0; i<4; i++) {
00791        out[i] = 0;
00792        in[i] = 0;
00793    }
00794    is_vectorin = 0;
00795    is_vectorout = 0;
00796    std::fill_n(arch_reg.src, MAX_REG_OPERANDS, -1);
00797    std::fill_n(arch_reg.dst, MAX_REG_OPERANDS, -1);
00798    pred = 0;

00800    ar2 = 0;
00801    space = m_space_spec;
00802    memory_op = no_memory_op;
00803    data_size = 0;
00804    if ( has_memory_read() || has_memory_write() ) {
00805       unsigned to_type = get_type();
00806       data_size = datatype2size(to_type);
00807       memory_op = has_memory_read() ? memory_load : memory_store;
00808    }
00809 
00810    bool has_dst = false ;
00811 
00812    switch ( get_opcode() ) {
00813 #define OP_DEF(OP,FUNC,STR,DST,CLASSIFICATION) case OP: has_dst = (DST!=0); break;
00814 #include "opcodes.def"
00815 #undef OP_DEF
00816    default:
00817       printf( "Execution error: Invalid opcode (0x%x)\n", get_opcode() );
00818       break;
00819    }
00820 
00821    switch( m_cache_option ) {
00822    case CA_OPTION: cache_op = CACHE_ALL; break;

00824    case CS_OPTION: cache_op = CACHE_STREAMING; break;
00825    case LU_OPTION: cache_op = CACHE_LAST_USE; break;
00826    case CV_OPTION: cache_op = CACHE_VOLATILE; break;
00827    case WB_OPTION: cache_op = CACHE_WRITE_BACK; break;

00829    default: 
00830       if( m_opcode == LD_OP || m_opcode == LDU_OP ) 
00831          cache_op = CACHE_ALL;
00832       else if( m_opcode == ST_OP ) 
00833          cache_op = CACHE_WRITE_BACK;
00834       else if( m_opcode == ATOM_OP ) 
00835          cache_op = CACHE_GLOBAL;
00836       break;
00837    }
00838 
00839    set_opcode_and_latency();
00840 
00841    // Get register operands
00842    int n=0,m=0;
00843    ptx_instruction::const_iterator opr=op_iter_begin();
00844    for ( ; opr != op_iter_end(); opr++, n++ ) { //process operands
00845       const operand_info &o = *opr;
00846       if ( has_dst && n==0 ) {
00847          // Do not set the null register "_" as an architectural register
00848          if ( o.is_reg() && !o.is_non_arch_reg() ) {
00849             out[0] = o.reg_num();
00850             arch_reg.dst[0] = o.arch_reg_num();
00851          } else if ( o.is_vector() ) {
00852             is_vectorin = 1;
00853             unsigned num_elem = o.get_vect_nelem();
00854             if( num_elem >= 1 ) out[0] = o.reg1_num();
00855             if( num_elem >= 2 ) out[1] = o.reg2_num();
00856             if( num_elem >= 3 ) out[2] = o.reg3_num();
00857             if( num_elem >= 4 ) out[3] = o.reg4_num();
00858             for (int i = 0; i < num_elem; i++) 
00859                arch_reg.dst[i] = o.arch_reg_num(i);
00860          }
00861       } else {
00862          if ( o.is_reg() && !o.is_non_arch_reg() ) {
00863             int reg_num = o.reg_num();
00864             arch_reg.src[m] = o.arch_reg_num();
00865             switch ( m ) {
00866             case 0: in[0] = reg_num; break;
00867             case 1: in[1] = reg_num; break;
00868             case 2: in[2] = reg_num; break;
00869             default: break; 
00870             }
00871             m++;
00872          } else if ( o.is_vector() ) {
00873             //assert(m == 0); //only support 1 vector operand (for textures) right now
00874             is_vectorout = 1;
00875             unsigned num_elem = o.get_vect_nelem();
00876             if( num_elem >= 1 ) in[0] = o.reg1_num();
00877             if( num_elem >= 2 ) in[1] = o.reg2_num();
00878             if( num_elem >= 3 ) in[2] = o.reg3_num();
00879             if( num_elem >= 4 ) in[3] = o.reg4_num();
00880             for (int i = 0; i < num_elem; i++) 
00881                arch_reg.src[i] = o.arch_reg_num(i);
00882             m+=4;
00883          }
00884       }
00885    }
00886 
00887    // Get predicate
00888    if(has_pred()) {
00889        const operand_info &p = get_pred();
00890        pred = p.reg_num();
00891    }
00892 
00893    // Get address registers inside memory operands.
00894    // Assuming only one memory operand per instruction,
00895    //  and maximum of two address registers for one memory operand.
00896    if( has_memory_read() || has_memory_write() ) {
00897       ptx_instruction::const_iterator op=op_iter_begin();
00898       for ( ; op != op_iter_end(); op++, n++ ) { //process operands
00899          const operand_info &o = *op;
00900

00902              // We do not support the null register as a memory operand

00907             if(o.is_memory_operand2()) {
00908 
00909                // memory operand with one address register (ex. g[$r1+0x4] or s[$r2+=0x4])
00910                if(o.get_double_operand_type() == 0 || o.get_double_operand_type() == 3){
00911                   ar1 = o.reg_num();

00914                }
00915                // memory operand with two address register (ex. s[$r1+$r1] or g[$r1+=$r2])
00916                else if(o.get_double_operand_type() == 1 || o.get_double_operand_type() == 2) {
00917                   ar1 = o.reg1_num();
00918                   arch_reg.src[4] = o.arch_reg_num();

00920                   arch_reg.src[5] = o.arch_reg_num();

00923             }

00925

00988       //

01053          size = (size<(p.get_size()/8))?size:(p.get_size()/8);
01054       } 
01055       // copy the parameter over word-by-word so that parameter that crosses a memory page can be copied over
01056       const size_t word_size = 4; 
01057       for (size_t idx = 0; idx < size; idx += word_size) {
01058          const char *pdata = reinterpret_cast<const char*>(param_value.pdata) + idx; // cast to char * for ptr arithmetic
01059          param_mem->write(param_address + idx, word_size, pdata,NULL,NULL); 
01060       }
01061       param->set_address(param_address);
01062       param_address += size; 
01063    }
01064 }

01163    const ptx_instruction *pI = m_func_info->get_instruction(pc);
01164    set_npc( pc + pI->inst_size() );
01165    
01166 
01167    try {
01168 
01169    clearRPC();
01170    m_last_set_operand_value.u64 = 0;
01171 
01172    if(is_done())
01173    {
01174       printf("attempted to execute instruction on a thread that is already done.\n");
01175       assert(0);
01176    }
01177    
01178    if ( g_debug_execution >= 6 || m_gpu->get_config().get_ptx_inst_debug_to_file()) {
01179       if ( (g_debug_thread_uid==0) || (get_uid() == (unsigned)g_debug_thread_uid) ) {

01186

01188       const operand_info &pred = pI->get_pred();

01190       if(pI->get_pred_mod() == -1) {
01191             skip = (pred_value.pred & 0x0001) ^ pI->get_pred_neg(); //ptxplus inverts the zero flag
01192       } else {
01193             skip = !pred_lookup(pI->get_pred_mod(), pred_value.pred & 0x000F);
01194       }
01195    }
01196    
01197    if( skip ) {
01198       inst.set_not_active(lane_id);
01199    } else {
01200       const ptx_instruction *pI_saved = pI;
01201       ptx_instruction *pJ = NULL;
01202       if( pI->get_opcode() == VOTE_OP ) {
01203          pJ = new ptx_instruction(*pI);
01204          *((warp_inst_t*)pJ) = inst; // copy active mask information
01205          pI = pJ;
01206       }
01207       switch ( pI->get_opcode() ) {
01208 #define OP_DEF(OP,FUNC,STR,DST,CLASSIFICATION) case OP: FUNC(pI,this); op_classification = CLASSIFICATION; break;
01209 #include "opcodes.def"
01210 #undef OP_DEF
01211       default: printf( "Execution error: Invalid opcode (0x%x)\n", pI->get_opcode() ); break;
01212       }
01213       delete pJ;
01214       pI = pI_saved;
01215       
01216       // Run exit instruction if exit option included
01217       if(pI->is_exit())
01218          exit_impl(pI,this);
01219    }
01220    
01221 
01222 
01223    const gpgpu_functional_sim_config &config = m_gpu->get_config();
01224    
01225    // Output instruction information to file and stdout
01226    if( config.get_ptx_inst_debug_to_file() != 0 && 
01227         (config.get_ptx_inst_debug_thread_uid() == 0 || config.get_ptx_inst_debug_thread_uid() == get_uid()) ) {

01229              "[thd=%u] : (%s:%u - %s)\n",
01230              get_uid(),
01231              pI->source_file(), pI->source_line(), pI->get_source() );
01232       //fprintf(ptx_inst_debug_file, "has memory read=%d, has memory write=%d\n", pI->has_memory_read(), pI->has_memory_write());
01233       fflush(m_gpu->get_ptx_inst_debug_file());
01234    }
01235 
01236    if ( ptx_debug_exec_dump_cond<5>(get_uid(), pc) ) {
01237       dim3 ctaid = get_ctaid();
01238       dim3 tid = get_tid();
01239       printf("%u [thd=%u][i=%u] : ctaid=(%u,%u,%u) tid=(%u,%u,%u) icount=%u [pc=%u] (%s:%u - %s)  [0x%llx]\n", 
01240              g_ptx_sim_num_insn, 
01241              get_uid(),
01242              pI->uid(), ctaid.x,ctaid.y,ctaid.z,tid.x,tid.y,tid.z,
01243              get_icount(),

01247    }

01255       insn_space = last_space();

01260    
01261    if ( pI->get_opcode() == ATOM_OP ) {

01263       insn_space = last_space();

01273    }
01274 
01275    // Output register information to file and stdout
01276    if( config.get_ptx_inst_debug_to_file()!=0 && 
01277        (config.get_ptx_inst_debug_thread_uid()==0||config.get_ptx_inst_debug_thread_uid()==get_uid()) ) {

01283       if ( ptx_debug_exec_dump_cond<6>(get_uid(), pc) )

01292    
01293    //not using it with functional simulation mode

01296    
01297    if ( gpgpu_ptx_instruction_classification ) {
01298       init_inst_classification_stat();
01299       unsigned space_type=0;
01300       switch ( pI->get_space().get_type() ) {
01301       case global_space: space_type = 10; break;
01302       case local_space:  space_type = 11; break; 
01303       case tex_space:    space_type = 12; break;

01316       StatAddSample( g_inst_op_classification_stat[g_ptx_kernel_count], (int)  pI->get_opcode() );
01317    }
01318    if ( (g_ptx_sim_num_insn % 100000) == 0 ) {
01319       dim3 ctaid = get_ctaid();
01320       dim3 tid = get_tid();
01321       printf("GPGPU-Sim PTX: %u instructions simulated : ctaid=(%u,%u,%u) tid=(%u,%u,%u)\n",
01322              g_ptx_sim_num_insn, ctaid.x,ctaid.y,ctaid.z,tid.x,tid.y,tid.z );
01323       fflush(stdout);
01324    }
01325    
01326    // "Return values"
01327    if(!skip) {
01328       inst.space = insn_space;
01329       inst.set_addr(lane_id, insn_memaddr);
01330       inst.data_size = insn_data_size; // simpleAtomicIntrinsics
01331       assert( inst.memory_op == insn_memory_op );
01332    } 
01333 
01334    } catch ( int x  ) {
01335       printf("GPGPU-Sim PTX: ERROR (%d) executing intruction (%s:%u)\n", x, pI->source_file(), pI->source_line() );
01336       printf("GPGPU-Sim PTX:       '%s'\n", pI->get_source() );
01337       abort();
01338    }
01339       
01340 }

01357 unsigned ptx_sim_init_thread( kernel_info_t &kernel,
01358                               ptx_thread_info** thread_info,
01359                               int sid,
01360                               unsigned tid,
01361                               unsigned threads_left,
01362                               unsigned num_threads, 
01363                               core_t *core, 
01364                               unsigned hw_cta_id, 
01365                               unsigned hw_warp_id,
01366                               gpgpu_t *gpu,
01367                               bool isInFunctionalSimulationMode)
01368 {
01369    std::list<ptx_thread_info *> &active_threads = kernel.active_threads();
01370 
01371    static std::map<unsigned,memory_space*> shared_memory_lookup;
01372    static std::map<unsigned,ptx_cta_info*> ptx_cta_lookup;
01373    static std::map<unsigned,std::map<unsigned,memory_space*> > local_memory_lookup;
01374 
01375    if ( *thread_info != NULL ) {
01376       ptx_thread_info *thd = *thread_info;
01377       assert( thd->is_done() );
01378       if ( g_debug_execution==-1 ) {
01379          dim3 ctaid = thd->get_ctaid();
01380          dim3 t = thd->get_tid();
01381          printf("GPGPU-Sim PTX simulator:  thread exiting ctaid=(%u,%u,%u) tid=(%u,%u,%u) uid=%u\n",
01382                 ctaid.x,ctaid.y,ctaid.z,t.x,t.y,t.z, thd->get_uid() );
01383          fflush(stdout);
01384       }
01385       thd->m_cta_info->register_deleted_thread(thd);
01386       delete thd;
01387       *thread_info = NULL;
01388    }
01389 
01390    if ( !active_threads.empty() ) {
01391       assert( active_threads.size() <= threads_left );
01392       ptx_thread_info *thd = active_threads.front(); 
01393       active_threads.pop_front();
01394       *thread_info = thd;
01395       thd->init(gpu, core, sid, hw_cta_id, hw_warp_id, tid, isInFunctionalSimulationMode );
01396       return 1;
01397    }
01398 
01399    if ( kernel.no_more_ctas_to_run() ) {
01400       return 0; //finished!
01401    }
01402 
01403    if ( threads_left < kernel.threads_per_cta() ) {
01404       return 0;
01405    }
01406 
01407    if ( g_debug_execution==-1 ) {
01408       printf("GPGPU-Sim PTX simulator:  STARTING THREAD ALLOCATION --> \n");
01409       fflush(stdout);
01410    }
01411 
01412    //initializing new CTA
01413    ptx_cta_info *cta_info = NULL;
01414    memory_space *shared_mem = NULL;
01415 
01416    unsigned cta_size = kernel.threads_per_cta();
01417    unsigned max_cta_per_sm = num_threads/cta_size; // e.g., 256 / 48 = 5 
01418    assert( max_cta_per_sm > 0 );
01419 
01420    unsigned sm_idx = (tid/cta_size)*gpgpu_param_num_shaders + sid;
01421 
01422    if ( shared_memory_lookup.find(sm_idx) == shared_memory_lookup.end() ) {
01423       if ( g_debug_execution >= 1 ) {
01424          printf("  <CTA alloc> : sm_idx=%u sid=%u max_cta_per_sm=%u\n", 
01425                 sm_idx, sid, max_cta_per_sm );
01426       }
01427       char buf[512];
01428       snprintf(buf,512,"shared_%u", sid);
01429       shared_mem = new memory_space_impl<16*1024>(buf,4);
01430       shared_memory_lookup[sm_idx] = shared_mem;
01431       cta_info = new ptx_cta_info(sm_idx);
01432       ptx_cta_lookup[sm_idx] = cta_info;
01433    } else {
01434       if ( g_debug_execution >= 1 ) {
01435          printf("  <CTA realloc> : sm_idx=%u sid=%u max_cta_per_sm=%u\n", 
01436                 sm_idx, sid, max_cta_per_sm );
01437       }
01438       shared_mem = shared_memory_lookup[sm_idx];
01439       cta_info = ptx_cta_lookup[sm_idx];
01440       cta_info->check_cta_thread_status_and_reset();
01441    }
01442 
01443    std::map<unsigned,memory_space*> &local_mem_lookup = local_memory_lookup[sid];
01444    while( kernel.more_threads_in_cta() ) {
01445       dim3 ctaid3d = kernel.get_next_cta_id();
01446       unsigned new_tid = kernel.get_next_thread_id();
01447       dim3 tid3d = kernel.get_next_thread_id_3d();
01448       kernel.increment_thread_id();
01449       new_tid += tid;
01450       ptx_thread_info *thd = new ptx_thread_info(kernel);
01451    
01452       memory_space *local_mem = NULL;
01453       std::map<unsigned,memory_space*>::iterator l = local_mem_lookup.find(new_tid);
01454       if ( l != local_mem_lookup.end() ) {
01455          local_mem = l->second;
01456       } else {
01457          char buf[512];
01458          snprintf(buf,512,"local_%u_%u", sid, new_tid);
01459          local_mem = new memory_space_impl<32>(buf,32);
01460          local_mem_lookup[new_tid] = local_mem;
01461       }
01462       thd->set_info(kernel.entry());
01463       thd->set_nctaid(kernel.get_grid_dim());
01464       thd->set_ntid(kernel.get_cta_dim());
01465       thd->set_ctaid(ctaid3d);
01466       thd->set_tid(tid3d);
01467       if( kernel.entry()->get_ptx_version().extensions() ) 
01468          thd->cpy_tid_to_reg(tid3d);
01469       thd->set_valid();
01470       thd->m_shared_mem = shared_mem;
01471       function_info *finfo = thd->func_info();
01472       symbol_table *st = finfo->get_symtab();
01473       thd->func_info()->param_to_shared(thd->m_shared_mem,st);
01474       thd->m_cta_info = cta_info;
01475       cta_info->add_thread(thd);
01476       thd->m_local_mem = local_mem;
01477       if ( g_debug_execution==-1 ) {
01478          printf("GPGPU-Sim PTX simulator:  allocating thread ctaid=(%u,%u,%u) tid=(%u,%u,%u) @ 0x%Lx\n",
01479                 ctaid3d.x,ctaid3d.y,ctaid3d.z,tid3d.x,tid3d.y,tid3d.z, (unsigned long long)thd );
01480          fflush(stdout);
01481       }
01482       active_threads.push_back(thd);
01483    }
01484    if ( g_debug_execution==-1 ) {
01485       printf("GPGPU-Sim PTX simulator:  <-- FINISHING THREAD ALLOCATION\n");
01486       fflush(stdout);
01487    }
01488 
01489    kernel.increment_cta_id();
01490 
01491    assert( active_threads.size() <= threads_left );
01492    *thread_info = active_threads.front();
01493    (*thread_info)->init(gpu, core, sid, hw_cta_id, hw_warp_id, tid,isInFunctionalSimulationMode );
01494    active_threads.pop_front();
01495    return 1;
01496 }
