00090    assert(sample >= 0);
00091    
00092    int bin;
00093    int v = sample;
00094    register unsigned int shift;
00095 
00096    bin =   (v > 0xFFFF) << 4; v >>= bin;
00097    shift = (v > 0xFF  ) << 3; v >>= shift; bin |= shift;
00098    shift = (v > 0xF   ) << 2; v >>= shift; bin |= shift;
00099    shift = (v > 0x3   ) << 1; v >>= shift; bin |= shift;
