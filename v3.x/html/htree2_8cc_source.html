00108   // input capacitance of a repeater  = input capacitance of nand.

00116       deviceType->Vth/deviceType->Vdd, deviceType->Vth/deviceType->Vdd, RISE);

00127     deviceType->Vdd * deviceType->Vdd * wire_bw ;

00238 /* calculates the input h-tree delay/power
00239  * A nand gate is used at each node to
00240  * limit the signal
00241  * The area of an unbalanced htree (rows != columns)
00242  * depends on how data is traversed.
00243  * In the following function, if ( no. of rows < no. of columns),
00244  * then data first traverse in excess hor. links until vertical
00245  * and horizontal nodes are same.
00246  * If no. of rows is bigger, then data traverse in
00247  * a hor. link followed by a ver. link in a repeated
00248  * fashion (similar to a balanced tree) until there are no
00249  * hor. links left. After this it goes through the remaining vertical
00250  * links.
00251  */

00257   double l_eff = 0;
00258   Wire *wtemp1 = 0, *wtemp2 = 0, *wtemp3 = 0;
00259   double len = 0, ht = 0;
00260   int option = 0;
00261 
00262   int h = (int) _log2(ndwl/2); // horizontal nodes
00263   int v = (int) _log2(ndbl/2); // vertical nodes
00264   double len_temp;
00265   double ht_temp;
00266   if (uca_tree)
00267   {//Sheng: this computation do not consider the wires that route from edge to middle.
00268     ht_temp = (mat_height*ndbl/2 +/* since uca_tree models interbank tree, mat_height => bank height */
00269         ((add_bits + data_in_bits + data_out_bits + (search_data_in_bits + search_data_out_bits)) * g_tp.wire_outside_mat.pitch *
00270          2 * (1-pow(0.5,h))))/2;
00271     len_temp = (mat_width*ndwl/2 +
00272         ((add_bits + data_in_bits + data_out_bits + (search_data_in_bits + search_data_out_bits)) * g_tp.wire_outside_mat.pitch *
00273          2 * (1-pow(0.5,v))))/2;
00274   }
00275   else
00276   {

00278       ht_temp = ((mat_height*ndbl/2) +
00279           ((add_bits + (search_data_in_bits + search_data_out_bits))* (ndbl/2-1) * g_tp.wire_outside_mat.pitch) +
00280           ((data_in_bits + data_out_bits) * g_tp.wire_outside_mat.pitch * h)
00281           )/2;

00283         ((add_bits + (search_data_in_bits + search_data_out_bits)) * (ndwl/2-1) * g_tp.wire_outside_mat.pitch) +
