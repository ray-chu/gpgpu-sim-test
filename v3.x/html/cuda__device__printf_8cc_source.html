00034 {
00035    FILE *fp = stdout;

00037    unsigned arg_offset=0;
00038    char buf[64];

00041       char c = fmtstr[i++];

00043          if( c != '%' ) {
00044             fprintf(fp,"%c",c);
00045          } else {
00046             in_fmt=true;

00048             j=1;
00049          }
00050       } else {
00051          if(!( c == 'u' || c == 'f' || c == 'd' )) {
00052             printf("GPGPU-Sim PTX: ERROR ** printf parsing support is limited to %%u, %%f, %%d at present");
00053             abort();

00055          buf[j] = c;
00056          buf[j+1] = 0;
00057          void* ptr = (void*)&arg_list[arg_offset];
00058          //unsigned long long value = ((unsigned long long*)arg_list)[arg_offset];
00059          if( c == 'u' || c == 'd' ) {
00060             fprintf(fp,buf,*((unsigned long long*)ptr));
00061          } else if( c == 'f' ) {
00062             double tmp = *((double*)ptr);
00063             fprintf(fp,buf,tmp);
00064          }
00065          arg_offset++;
00066          in_fmt=false;
00067       }
