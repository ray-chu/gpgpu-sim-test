00207     double h_tr_region  = fold_dimension - 2 * g_tp.HPOWERRAIL;
00208     // TODO : w_folded_tr must come from Component::compute_gate_area()
00209     double ratio_p_to_n = 2.0 / (2.0 + 1.0);
00210     if (nchannel)

00217     }
00218   }

00220 
00221   if (num_folded_tr < 2)
00222   {
00223     w_folded_tr = width;
00224   }
00225 
00226   double total_drain_w = (g_tp.w_poly_contact + 2 * g_tp.spacing_poly_to_contact) +  // only for drain
00227                          (stack - 1) * g_tp.spacing_poly_to_poly;
00228   double drain_h_for_sidewall = w_folded_tr;
00229   double total_drain_height_for_cap_wrt_gate = w_folded_tr + 2 * w_folded_tr * (stack - 1);
00230   if (num_folded_tr > 1)
00231   {
00232     total_drain_w += (num_folded_tr - 2) * (g_tp.w_poly_contact + 2 * g_tp.spacing_poly_to_contact) +
00233                      (num_folded_tr - 1) * ((stack - 1) * g_tp.spacing_poly_to_poly);
00234 
00235     if (num_folded_tr%2 == 0)
00236     {
00237       drain_h_for_sidewall = 0;
00238     }
00239     total_drain_height_for_cap_wrt_gate *= num_folded_tr;
00240     drain_C_metal_connecting_folded_tr   = g_tp.wire_local.C_per_um * total_drain_w;
00241   }
00242 
00243   double drain_C_area     = c_junc_area * total_drain_w * w_folded_tr;
00244   double drain_C_sidewall = c_junc_sidewall * (drain_h_for_sidewall + 2 * total_drain_w);
00245   double drain_C_wrt_gate = (c_fringe + c_overlap) * total_drain_height_for_cap_wrt_gate;
00246 
00247   return (drain_C_area + drain_C_sidewall + drain_C_wrt_gate + drain_C_metal_connecting_folded_tr);

00336 // "Timing Models for MOS Circuits" by Mark Horowitz, 1984
00337 double horowitz(
00338     double inputramptime, // input rise time
00339     double tf,            // time constant of gate
00340     double vs1,           // threshold voltage
00341     double vs2,           // threshold voltage
00342     int    rise)          // whether input rises or fall
00343 {
00344   if (inputramptime == 0 && vs1 == vs2)
00345   {
00346     return tf * (vs1 < 1 ? -log(vs1) : log(vs1));
00347   }
00348   double a, b, td;
00349 
00350   a = inputramptime / tf;
00351   if (rise == RISE)
00352   {
00353     b = 0.5;
00354     td = tf * sqrt(log(vs1)*log(vs1) + 2*a*b*(1.0 - vs1)) + tf*(log(vs1) - log(vs2));
00355   }
00356   else
00357   {
00358     b = 0.4;
00359     td = tf * sqrt(log(1.0 - vs1)*log(1.0 - vs1) + 2*a*b*(vs1)) + tf*(log(1.0 - vs1) - log(1.0 - vs2));
00360   }
00361   return (td);
00362 }
