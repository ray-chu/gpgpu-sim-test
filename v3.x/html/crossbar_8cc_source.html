00091 void Crossbar::compute_power()
00092 {
00093 
00094   Wire winit(4, 4);
00095   double tri_cap = output_buffer();
00096   assert(tri_cap > 0);
00097   //area of a tristate logic
00098   double g_area = compute_gate_area(INV, 1, TriS2*g_tp.min_w_nmos_, TriS2*min_w_pmos, g_tp.cell_h_def);
00099   g_area *= 2; // to model area of output transistors
00100   g_area += compute_gate_area (NAND, 2, TriS1*2*g_tp.min_w_nmos_, TriS1*min_w_pmos, g_tp.cell_h_def);
00101   g_area += compute_gate_area (NOR, 2, TriS1*g_tp.min_w_nmos_, TriS1*2*min_w_pmos, g_tp.cell_h_def);
00102   double width /*per tristate*/ = g_area/(CB_ADJ * g_tp.cell_h_def);
00103   // effective no. of tristate buffers that need to be laid side by side
00104   int ntri = (int)ceil(g_tp.cell_h_def/(g_tp.wire_outside_mat.pitch));
00105   double wire_len = MAX(width*ntri*n_out, flit_size*g_tp.wire_outside_mat.pitch*n_out);
00106   Wire w1(g_ip->wt, wire_len);
00107 
00108   area.w = wire_len;
00109   area.h = g_tp.wire_outside_mat.pitch*n_inp*flit_size * CB_ADJ;
00110   Wire w2(g_ip->wt, area.h);
00111 
00112   double aspect_ratio_cb = (area.h/area.w)*(n_out/n_inp);
00113   if (aspect_ratio_cb > 1) aspect_ratio_cb = 1/aspect_ratio_cb;
00114 
00115   if (aspect_ratio_cb < ASPECT_THRESHOLD) {
00116     if (n_out > 2 && n_inp > 2) {
00117       CB_ADJ+=0.2;
00118       //cout << "CB ADJ " << CB_ADJ << endl;
00119       if (CB_ADJ < 4) {
00120         this->compute_power();
00121       }
00122     }
00123   }
00124 
00125 
00126 
00127   power.readOp.dynamic = (w1.power.readOp.dynamic + w2.power.readOp.dynamic + (tri_inp_cap * n_out + tri_out_cap * n_inp + tri_ctr_cap + tri_int_cap) * Vdd*Vdd)*flit_size;

00129     cmos_Isub_leakage(g_tp.min_w_nmos_*TriS2*2, min_w_pmos*TriS2*2, 1, inv) *Vdd+
00130     cmos_Isub_leakage(g_tp.min_w_nmos_*TriS1*3, min_w_pmos*TriS1*3, 2, nand)*Vdd+
00131     cmos_Isub_leakage(g_tp.min_w_nmos_*TriS1*3, min_w_pmos*TriS1*3, 2, nor) *Vdd+
00132     w1.power.readOp.leakage + w2.power.readOp.leakage);
00133   power.readOp.gate_leakage = n_inp * n_out * flit_size * (
00134       cmos_Ig_leakage(g_tp.min_w_nmos_*TriS2*2, min_w_pmos*TriS2*2, 1, inv) *Vdd+

00138 
00139   // delay calculation
00140   double l_eff = n_inp*flit_size*g_tp.wire_outside_mat.pitch;
