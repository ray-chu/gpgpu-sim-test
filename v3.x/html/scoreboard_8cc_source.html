00137 {
00138     // Get list of all input and output registers
00139     std::set<int> inst_regs;
00140 
00141     if(inst->out[0] > 0) inst_regs.insert(inst->out[0]);
00142     if(inst->out[1] > 0) inst_regs.insert(inst->out[1]);
00143     if(inst->out[2] > 0) inst_regs.insert(inst->out[2]);
00144     if(inst->out[3] > 0) inst_regs.insert(inst->out[3]);
00145     if(inst->in[0] > 0) inst_regs.insert(inst->in[0]);
00146     if(inst->in[1] > 0) inst_regs.insert(inst->in[1]);
00147     if(inst->in[2] > 0) inst_regs.insert(inst->in[2]);
00148     if(inst->in[3] > 0) inst_regs.insert(inst->in[3]);
00149     if(inst->pred > 0) inst_regs.insert(inst->pred);
00150     if(inst->ar1 > 0) inst_regs.insert(inst->ar1);
00151     if(inst->ar2 > 0) inst_regs.insert(inst->ar2);
00152 
00153     // Check for collision, get the intersection of reserved registers and instruction registers
00154     std::set<int>::const_iterator it2;
00155     for ( it2=inst_regs.begin() ; it2 != inst_regs.end(); it2++ )
00156         if(reg_table[wid].find(*it2) != reg_table[wid].end()) {
00157             return true;
00158         }
