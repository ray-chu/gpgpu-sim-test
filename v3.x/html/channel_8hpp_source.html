00101 void Channel<T>::ReadInputs() {
00102   if(_input) {
00103     _wait_queue.push(make_pair(GetSimTime() + _delay - 1, _input));

00108 template<typename T>
00109 void Channel<T>::WriteOutputs() {
00110   _output = 0;
00111   if(_wait_queue.empty()) {
00112     return;
00113   }
00114   pair<int, T *> const & item = _wait_queue.front();
00115   int const & time = item.first;
00116   if(GetSimTime() < time) {
00117     return;
00118   }
00119   assert(GetSimTime() == time);

00121   assert(_output);
00122   _wait_queue.pop();
