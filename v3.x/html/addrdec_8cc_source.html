00068 { 
00069    if (!gap) {
00070       return addrdec_packbits( ~(addrdec_mask[CHIP] | sub_partition_id_mask), addr, 64, 0 ); 
00071    } else {
00072       // see addrdec_tlx for explanation 
00073       unsigned long long int partition_addr; 
00074       partition_addr = ( (addr>>ADDR_CHIP_S) / m_n_channel) << ADDR_CHIP_S;

00082 void linear_to_raw_address_translation::addrdec_tlx(new_addr_type addr, addrdec_t *tlx) const
00083 {  
00084    unsigned long long int addr_for_chip,rest_of_addr;
00085    if (!gap) {
00086       tlx->chip = addrdec_packbits(addrdec_mask[CHIP], addr, addrdec_mkhigh[CHIP], addrdec_mklow[CHIP]);
00087       tlx->bk   = addrdec_packbits(addrdec_mask[BK], addr, addrdec_mkhigh[BK], addrdec_mklow[BK]);
00088       tlx->row  = addrdec_packbits(addrdec_mask[ROW], addr, addrdec_mkhigh[ROW], addrdec_mklow[ROW]);
00089       tlx->col  = addrdec_packbits(addrdec_mask[COL], addr, addrdec_mkhigh[COL], addrdec_mklow[COL]);
00090       tlx->burst= addrdec_packbits(addrdec_mask[BURST], addr, addrdec_mkhigh[BURST], addrdec_mklow[BURST]);
00091    } else {
00092       // Split the given address at ADDR_CHIP_S into (MSBs,LSBs)
00093       // - extract chip address using modulus of MSBs
00094       // - recreate the rest of the address by stitching the quotient of MSBs and the LSBs 
00095       addr_for_chip = (addr>>ADDR_CHIP_S) % m_n_channel; 
00096       rest_of_addr = ( (addr>>ADDR_CHIP_S) / m_n_channel) << ADDR_CHIP_S; 
00097       rest_of_addr |= addr & ((1 << ADDR_CHIP_S) - 1); 
00098 
00099       tlx->chip = addr_for_chip; 
00100       tlx->bk   = addrdec_packbits(addrdec_mask[BK], rest_of_addr, addrdec_mkhigh[BK], addrdec_mklow[BK]);
00101       tlx->row  = addrdec_packbits(addrdec_mask[ROW], rest_of_addr, addrdec_mkhigh[ROW], addrdec_mklow[ROW]);
00102       tlx->col  = addrdec_packbits(addrdec_mask[COL], rest_of_addr, addrdec_mkhigh[COL], addrdec_mklow[COL]);
00103       tlx->burst= addrdec_packbits(addrdec_mask[BURST], rest_of_addr, addrdec_mkhigh[BURST], addrdec_mklow[BURST]);
00104    }
00105 
00106    // combine the chip address and the lower bits of DRAM bank address to form the subpartition ID
00107    unsigned sub_partition_addr_mask = m_n_sub_partition_in_channel - 1; 
00108    tlx->sub_partition = tlx->chip * m_n_sub_partition_in_channel
00109                         + (tlx->bk & sub_partition_addr_mask); 
00110 }

00128 
00129    addrdec_mask[CHIP] = 0x0;
00130    addrdec_mask[BK]   = 0x0;

00133    addrdec_mask[BURST]= 0x0;

00135    int ofs = 63;

00138          case 'D': case 'd':  
00139             assert(dramid_parsed != 1); addrdec_mask[CHIP]  |= (1ULL << ofs); ofs--; break;
00140          case 'B': case 'b':   addrdec_mask[BK]    |= (1ULL << ofs); ofs--; break;
00141          case 'R': case 'r':   addrdec_mask[ROW]   |= (1ULL << ofs); ofs--; break;
00142          case 'C': case 'c':   addrdec_mask[COL]   |= (1ULL << ofs); ofs--; break;
00143          case 'S': case 's':   addrdec_mask[BURST] |= (1ULL << ofs); addrdec_mask[COL]   |= (1ULL << ofs); ofs--; break;
00144          // ignore bit
00145          case '0': ofs--; break;
00146          // ignore character
00147          case '|':
00148          case ' ':
00149          case '.': break;
00150          default:
00151             fprintf(stderr, "ERROR: Invalid address mapping character '%c' in option '%s'\n", *cmapping, option);
00152       }
00153       cmapping += 1;
00154    }
00155 
00156    if (ofs != -1) {
00157       fprintf(stderr, "ERROR: Invalid address mapping length (%d) in option '%s'\n", 63 - ofs, option);
