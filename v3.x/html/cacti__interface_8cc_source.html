00069 void uca_org_t::find_delay()
00070 {
00071   mem_array * data_arr = data_array2;
00072   mem_array * tag_arr  = tag_array2;
00073 
00074   // check whether it is a regular cache or scratch ram
00075   if (g_ip->pure_ram|| g_ip->pure_cam || g_ip->fully_assoc)
00076   {
00077     access_time = data_arr->access_time;
00078   }
00079   // Both tag and data lookup happen in parallel
00080   // and the entire set is sent over the data array h-tree without
00081   // waiting for the way-select signal --TODO add the corresponding
00082   // power overhead Nav
00083   else if (g_ip->fast_access == true)
00084   {
00085     access_time = MAX(tag_arr->access_time, data_arr->access_time);
00086   }
00087   // Tag is accessed first. On a hit, way-select signal along with the
00088   // address is sent to read/write the appropriate block in the data
00089   // array
00090   else if (g_ip->is_seq_acc == true)
00091   {
00092     access_time = tag_arr->access_time + data_arr->access_time;
00093   }
00094   // Normal access: tag array access and data array access happen in parallel.
00095   // But, the data array will wait for the way-select and transfer only the
00096   // appropriate block over the h-tree.
00097   else
00098   {
00099     access_time = MAX(tag_arr->access_time + data_arr->delay_senseamp_mux_decoder,
00100                       data_arr->delay_before_subarray_output_driver) +
00101                   data_arr->delay_from_subarray_output_driver_to_output;
00102   }
00103 }
