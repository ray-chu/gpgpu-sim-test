00074 void print_ptx_file( const char *p, unsigned source_num, const char *filename )
00075 {
00076    printf("\nGPGPU-Sim PTX: file _%u.ptx contents:\n\n", source_num );
00077    char *s = strdup(p);
00078    char *t = s;
00079    unsigned n=1;
00080    while ( *t != '\0'  ) {
00081       char *u = t;
00082       while ( (*u != '\n') && (*u != '\0') ) u++;
00083       unsigned last = (*u == '\0');
00084       *u = '\0';
00085       const ptx_instruction *pI = ptx_instruction_lookup(filename,n);
00086       char pc[64];
00087       if( pI && pI->get_PC() )
00088          snprintf(pc,64,"%4u", pI->get_PC() );
00089       else 
00090          snprintf(pc,64,"    ");
00091       printf("    _%u.ptx  %4u (pc=%s):  %s\n", source_num, n, pc, t );
00092       if ( last ) break;
00093       t = u+1;
00094       n++;
00095    }
00096    free(s);
00097    fflush(stdout);
