00059   register int t,s,j;
00060   double u[KK+KK-1];
00061   double ulp=(1.0/(1L<<30))/(1L<<22);               /* 2 to the -52 */
00062   double ss=2.0*ulp*((seed&0x3fffffff)+2);

00064   for (j=0;j<KK;j++) {
00065     u[j]=ss;                                /* bootstrap the buffer */
00066     ss+=ss; if (ss>=1.0) ss-=1.0-2*ulp;  /* cyclic shift of 51 bits */
00067   }
00068   u[1]+=ulp;                     /* make u[1] (and only u[1]) "odd" */

00070     for (j=KK-1;j>0;j--)
00071       u[j+j]=u[j],u[j+j-1]=0.0;                         /* "square" */

00073       u[j-(KK-LL)]=mod_sum(u[j-(KK-LL)],u[j]);
00074       u[j-KK]=mod_sum(u[j-KK],u[j]);
00075     }
00076     if (is_odd(s)) {                             /* "multiply by z" */
00077       for (j=KK;j>0;j--) u[j]=u[j-1];
00078       u[0]=u[KK];                    /* shift the buffer cyclically */
00079       u[LL]=mod_sum(u[LL],u[KK]);
00080     }
00081     if (s) s>>=1; else t--;
00082   }
00083   for (j=0;j<LL;j++) ran_u[j+KK-LL]=u[j];
