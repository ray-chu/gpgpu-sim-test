00388 void gpgpu_sim_wrapper::power_metrics_calculations()
00389 {
00390     total_sample_count++;
00391     kernel_sample_count++;
00392 
00393     // Current sample power
00394     double sample_power = proc->rt_power.readOp.dynamic + sample_cmp_pwr[CONST_DYNAMICP];
00395 
00396     // Average power
00397     // Previous + new + constant dynamic power (e.g., dynamic clocking power)
00398     kernel_tot_power += sample_power;
00399     kernel_power.avg = kernel_tot_power / kernel_sample_count;
00400     for(unsigned ind=0; ind<num_pwr_cmps; ++ind){
00401         kernel_cmp_pwr[ind].avg += (double)sample_cmp_pwr[ind];
00402     }
00403 
00404     for(unsigned ind=0; ind<num_perf_counters; ++ind){
00405         kernel_cmp_perf_counters[ind].avg += (double)sample_perf_counters[ind];
00406     }
00407 
00408     // Max Power
00409     if(sample_power > kernel_power.max){
00410         kernel_power.max = sample_power;
00411         for(unsigned ind=0; ind<num_pwr_cmps; ++ind){
00412             kernel_cmp_pwr[ind].max = (double)sample_cmp_pwr[ind];
00413         }
00414         for(unsigned ind=0; ind<num_perf_counters; ++ind){
00415             kernel_cmp_perf_counters[ind].max = sample_perf_counters[ind];
00416         }
00417 
00418     }
00419 
00420     // Min Power
00421     if(sample_power < kernel_power.min ||(kernel_power.min==0) ){
00422       kernel_power.min = sample_power;
00423       for(unsigned ind=0; ind<num_pwr_cmps; ++ind){
00424           kernel_cmp_pwr[ind].min = (double)sample_cmp_pwr[ind];
00425       }
00426       for(unsigned ind=0; ind<num_perf_counters; ++ind){
00427           kernel_cmp_perf_counters[ind].min = sample_perf_counters[ind];
00428       }
00429     }
00430 
00431     gpu_tot_power.avg = (gpu_tot_power.avg + sample_power);
00432     gpu_tot_power.max = (sample_power > gpu_tot_power.max) ? sample_power : gpu_tot_power.max;
00433     gpu_tot_power.min = ((sample_power < gpu_tot_power.min) || (gpu_tot_power.min == 0)) ? sample_power : gpu_tot_power.min;
00434 
00435 }

00694 {

00697         steady_state_tacking_file = gzopen(g_steady_state_tracking_filename,"a");

00709                     samples_counter.push_back(sample_perf_counters[i]);

00711

00713                     pwr_counter.push_back(sample_cmp_pwr[i]);

00715                 assert(pwr_counter.size() == (double)num_pwr_cmps);
00716                 assert(samples_counter.size() == (double)num_perf_counters);
00717             }else{
00718                 // Get current average
00719                 double temp_avg = sample_val / (double)samples.size() ;
00720 
00721                 if( abs(proc->rt_power.readOp.dynamic-temp_avg) < gpu_steady_power_deviation){ // Value is within threshold
00722                     sample_val += proc->rt_power.readOp.dynamic;
00723                     samples.push_back(proc->rt_power.readOp.dynamic);
00724                     for(unsigned i=0; i<(num_perf_counters); ++i){
00725                         samples_counter.at(i) += sample_perf_counters[i];
00726                     }
00727 
00728                     for(unsigned i=0; i<(num_pwr_cmps); ++i){
00729                         pwr_counter.at(i) += sample_cmp_pwr[i];

00731 
00732                 }else{  // Value exceeds threshold, not considered steady state
