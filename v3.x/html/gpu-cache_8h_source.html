00139     {
00140         cache_status= status;
00141         assert( config );
00142         char rp, wp, ap, mshr_type, wap;
00143 
00144         int ntok = sscanf(config,"%u:%u:%u,%c:%c:%c:%c,%c:%u:%u,%u:%u,%u",
00145                           &m_nset, &m_line_sz, &m_assoc, &rp, &wp, &ap, &wap,
00146                           &mshr_type, &m_mshr_entries,&m_mshr_max_merge,
00147                           &m_miss_queue_size,&m_result_fifo_entries,

00149 
00150         if ( ntok < 11 ) {
00151             if ( !strcmp(config,"none") ) {
00152                 m_disabled = true;
00153                 return;
00154             }
00155             exit_parse_error();
00156         }
00157         switch (rp) {
00158         case 'L': m_replacement_policy = LRU; break;
00159         case 'F': m_replacement_policy = FIFO; break;
00160         default: exit_parse_error();
00161         }
00162         switch (wp) {
00163         case 'R': m_write_policy = READ_ONLY; break;
00164         case 'B': m_write_policy = WRITE_BACK; break;
00165         case 'T': m_write_policy = WRITE_THROUGH; break;
00166         case 'E': m_write_policy = WRITE_EVICT; break;
00167         case 'L': m_write_policy = LOCAL_WB_GLOBAL_WT; break;
00168         default: exit_parse_error();
00169         }
00170         switch (ap) {
00171         case 'm': m_alloc_policy = ON_MISS; break;
00172         case 'f': m_alloc_policy = ON_FILL; break;
00173         default: exit_parse_error();
00174         }
00175         switch (mshr_type) {
00176         case 'F': m_mshr_type = TEX_FIFO; assert(ntok==12); break;
00177         case 'A': m_mshr_type = ASSOC; break;
00178         default: exit_parse_error();
00179         }
00180         m_line_sz_log2 = LOGB2(m_line_sz);
00181         m_nset_log2 = LOGB2(m_nset);
00182         m_valid = true;
00183 
00184         switch(wap){
00185         case 'W': m_write_alloc_policy = WRITE_ALLOCATE; break;
00186         case 'N': m_write_alloc_policy = NO_WRITE_ALLOCATE; break;
00187         default: exit_parse_error();
00188         }
00189 
00190         // detect invalid configuration 
00191         if (m_alloc_policy == ON_FILL and m_write_policy == WRITE_BACK) {
00192             // A writeback cache with allocate-on-fill policy will inevitably lead to deadlock:  
00193             // The deadlock happens when an incoming cache-fill evicts a dirty
00194             // line, generating a writeback request.  If the memory subsystem
00195             // is congested, the interconnection network may not have
00196             // sufficient buffer for the writeback request.  This stalls the
00197             // incoming cache-fill.  The stall may propagate through the memory
00198             // subsystem back to the output port of the same core, creating a
00199             // deadlock where the wrtieback request and the incoming cache-fill
00200             // are stalling each other.  
00201             assert(0 && "Invalid cache configuration: Writeback cache cannot allocate new line on fill. "); 
00202         }
00203 
00204         // default: port to data array width and granularity = line size 
00205         if (m_data_port_width == 0) {
00206             m_data_port_width = m_line_sz; 
00207         }
00208         assert(m_line_sz % m_data_port_width == 0); 
00209     }
