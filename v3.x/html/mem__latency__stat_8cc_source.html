00137 unsigned memory_stats_t::memlatstat_done(mem_fetch *mf )
00138 {
00139    unsigned mf_latency;
00140    mf_latency = (gpu_sim_cycle+gpu_tot_sim_cycle) - mf->get_timestamp();
00141    mf_num_lat_pw++;
00142    mf_tot_lat_pw += mf_latency;
00143    unsigned idx = LOGB2(mf_latency);
00144    assert(idx<32);
00145    mf_lat_table[idx]++;
00146    shader_mem_lat_log(mf->get_sid(), mf_latency);
00147    mf_total_lat_table[mf->get_tlx_addr().chip][mf->get_tlx_addr().bk] += mf_latency;
00148    if (mf_latency > max_mf_latency)
00149       max_mf_latency = mf_latency;
00150    return mf_latency;
