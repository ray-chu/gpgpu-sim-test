00055 void Wavefront::Allocate( )
00056 {
00057 
00058   int first_diag = -1;
00059 
00060   if(_num_requests == 0)
00061 
00062     // bypass allocator completely if there were no requests
00063     return;
00064   
00065   if(_num_requests == 1) {
00066 
00067     // if we only had a single request, we can immediately grant it
00068     _inmatch[_last_in] = _last_out;
00069     _outmatch[_last_out] = _last_in;
00070     first_diag = _last_in + _last_out;
00071 
00072   } else {
00073 
00074     // otherwise we have to loop through the diagonals of request matrix
00075 
00076     for(set<pair<int, int> >::const_reverse_iterator iter = 
00077       _priorities.rbegin();
00078     iter != _priorities.rend(); ++iter) {
00079       
00080       for ( int p = 0; p < _square; ++p ) {
00081     for ( int output = 0; output < _square; ++output ) {
00082       int input = ( ( _pri + p ) + ( _square - output ) ) % _square;
00083       if ( ( input < _inputs ) && ( output < _outputs ) && 
00084            ( _inmatch[input] == -1 ) && ( _outmatch[output] == -1 ) &&
00085            ( _request[input][output].label != -1 ) &&
00086            ( _request[input][output].in_pri == iter->second ) &&
00087            ( _request[input][output].out_pri == iter->first ) ) {
00088         // Grant!
00089         _inmatch[input] = output;
00090         _outmatch[output] = input;
00091         if(first_diag < 0) {
00092           first_diag = input + output;
00093         }
00094       }
00095     }
00096       }
00097     }
00098   }
00099 
00100   _num_requests = 0;
00101   _last_in = -1;
00102   _last_out = -1;
00103   _priorities.clear();
00104   
00105   assert(first_diag >= 0);
00106 
00107   // Round-robin the priority diagonal
00108   _pri = ( ( _skip_diags ? first_diag : _pri ) + 1 ) % _square;
00109 }
