00250 double Power_Module::powerWordLine(double memoryWidth, double memoryDepth){
00251   // wordline capacitance
00252   double Ccell = 2 * ( 4.0 * LAMBDA ) * Cg_pwr +  6 * MetalPitch * Cw ;     
00253   double Cwl = memoryWidth * Ccell ; 
00254 
00255   // wordline circuits
00256   double Warray = 8 * MetalPitch + memoryDepth ;
00257   double x = 1.0 + (5.0/16.0) * (1 + Co/Ci)  ;
00258   double Cpredecode = x * (Cw * Warray  * Ci) ;
00259   double Cdecode    = x * Cwl ;
00260 
00261   // bitline circuits
00262   double Harray =  6 * memoryWidth * MetalPitch ;
00263   double y = (1 + 0.25) * (1 + Co/Ci) ;
00264   double Cprecharge = y * ( Cw * Harray + 3 * channel_width * Ci ) ;
00265   double Cwren      = y * ( Cw * Harray + 2 * channel_width * Ci ) ;
00266 
00267   double Cbd = Cprecharge + Cwren ;
00268   double Cwd = 2 * Cpredecode + Cdecode ;

00270   return ( Cbd + Cwd ) * Vdd * Vdd * fCLK ;
