03683 float reduce_precision( float x, unsigned bits )
03684 {
03685    intfloat tmp;
03686    tmp.b = x;
03687    int v = tmp.a;
03688    int man = v & ((1<<23)-1);
03689    int mask =  ((1<<bits)-1) << (23-bits);
03690    int nv = (v & ((-1)-((1<<23)-1))) | (mask&man);
03691    tmp.a = nv;
03692    float result = tmp.b;
03693    return result;
03694 }

03734 float textureNormalizeElementSigned(int element, int bits)
03735 {
03736    if (bits) {
03737       int maxN = (1 << bits) - 1; 
03738       // removing upper bits 
03739       element &= maxN;
03740       // normalizing the number to [-1.0,1.0]
03741       maxN >>= 1;
03742       float output = (float) element / maxN;  
03743       if (output < -1.0f) output = -1.0f; 
03744       return output; 
03745    } else {
03746       return 0.0f; 
03747    }
03748 }

03750 float textureNormalizeElementUnsigned(unsigned int element, int bits)
03751 {
03752    if (bits) {
03753       unsigned int maxN = (1 << bits) - 1; 
03754       // removing upper bits and normalizing the number to [0.0,1.0]
03755       return (float)(element & maxN) / maxN;  
03756    } else {
03757       return 0.0f; 
03758    }
03759 }

04100       int lane_id = thread->get_hw_tid() % pI->warp_size();

04125             const operand_info &dst = pI->dst();
