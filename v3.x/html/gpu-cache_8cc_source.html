00109 enum cache_request_status tag_array::probe( new_addr_type addr, unsigned &idx ) const {
00110     //assert( m_config.m_write_policy == READ_ONLY );
00111     unsigned set_index = m_config.set_index(addr);
00112     new_addr_type tag = m_config.tag(addr);
00113 
00114     unsigned invalid_line = (unsigned)-1;
00115     unsigned valid_line = (unsigned)-1;
00116     unsigned valid_timestamp = (unsigned)-1;
00117 
00118     bool all_reserved = true;
00119 
00120     // check for hit or pending hit
00121     for (unsigned way=0; way<m_config.m_assoc; way++) {
00122         unsigned index = set_index*m_config.m_assoc+way;
00123         cache_block_t *line = &m_lines[index];
00124         if (line->m_tag == tag) {
00125             if ( line->m_status == RESERVED ) {
00126                 idx = index;
00127                 return HIT_RESERVED;
00128             } else if ( line->m_status == VALID ) {
00129                 idx = index;
00130                 return HIT;
00131             } else if ( line->m_status == MODIFIED ) {
00132                 idx = index;
00133                 return HIT;
00134             } else {
00135                 assert( line->m_status == INVALID );
00136             }
00137         }
00138         if (line->m_status != RESERVED) {
00139             all_reserved = false;
00140             if (line->m_status == INVALID) {
00141                 invalid_line = index;
00142             } else {
00143                 // valid line : keep track of most appropriate replacement candidate
00144                 if ( m_config.m_replacement_policy == LRU ) {
00145                     if ( line->m_last_access_time < valid_timestamp ) {
00146                         valid_timestamp = line->m_last_access_time;
00147                         valid_line = index;
00148                     }
00149                 } else if ( m_config.m_replacement_policy == FIFO ) {
00150                     if ( line->m_alloc_time < valid_timestamp ) {
00151                         valid_timestamp = line->m_alloc_time;
00152                         valid_line = index;
00153                     }
00154                 }
00155             }
00156         }
00157     }
00158     if ( all_reserved ) {
00159         assert( m_config.m_alloc_policy == ON_MISS ); 
00160         return RESERVATION_FAIL; // miss and not enough space in cache to allocate on miss
00161     }
00162 
00163     if ( invalid_line != (unsigned)-1 ) {
00164         idx = invalid_line;
00165     } else if ( valid_line != (unsigned)-1) {
00166         idx = valid_line;
00167     } else abort(); // if an unreserved block exists, it is either invalid or replaceable 
00168 
00169     return MISS;
00170 }

00185     enum cache_request_status status = probe(addr,idx);
00186     switch (status) {

01101 void tex_cache::cycle(){
01102     // send next request to lower level of memory
01103     if ( !m_request_fifo.empty() ) {
01104         mem_fetch *mf = m_request_fifo.peek();
01105         if ( !m_memport->full(mf->get_ctrl_size(),false) ) {
01106             m_request_fifo.pop();
01107             m_memport->push(mf);
01108         }
01109     }
01110     // read ready lines from cache
01111     if ( !m_fragment_fifo.empty() && !m_result_fifo.full() ) {
01112         const fragment_entry &e = m_fragment_fifo.peek();
01113         if ( e.m_miss ) {
01114             // check head of reorder buffer to see if data is back from memory
01115             unsigned rob_index = m_rob.next_pop_index();
01116             const rob_entry &r = m_rob.peek(rob_index);
01117             assert( r.m_request == e.m_request );
01118             assert( r.m_block_addr == m_config.block_addr(e.m_request->get_addr()) );
01119             if ( r.m_ready ) {
01120                 assert( r.m_index == e.m_cache_index );
01121                 m_cache[r.m_index].m_valid = true;
01122                 m_cache[r.m_index].m_block_addr = r.m_block_addr;
01123                 m_result_fifo.push(e.m_request);
01124                 m_rob.pop();
01125                 m_fragment_fifo.pop();
01126             }
01127         } else {
01128             // hit:
01129             assert( m_cache[e.m_cache_index].m_valid );
01130             assert( m_cache[e.m_cache_index].m_block_addr
01131                 == m_config.block_addr(e.m_request->get_addr()) );
01132             m_result_fifo.push( e.m_request );
01133             m_fragment_fifo.pop();
01134         }
01135     }
01136 }
